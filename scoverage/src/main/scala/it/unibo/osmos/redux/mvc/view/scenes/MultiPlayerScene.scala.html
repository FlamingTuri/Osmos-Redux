108   /** This method sends a request to enter the lobby
109     *
110     * @param user      the user
111     * @param levelInfo the level info, which may be not present if the user is a client
112     */
113   private def goToLobby(user: User, levelInfo: Option[LevelInfo]): Unit = {
114     /** We create the lobby context */
115     val lobbyContext = LobbyContext()
116 
117     /** We ask to enter in the lobby */
118     listener.onLobbyRequest(user, levelInfo, lobbyContext, onLobbyEnterResult)
119   }
120 
121   /** Result parsing function.
122     *
123     * @return a function which will send the user to the MultiPlayerLobbyScene if the result is true, showing an error otherwise
124     */
125   private def onLobbyEnterResult: (User, Option[LevelInfo], LobbyContext, GenericResponse[Boolean]) => Unit = (user, levelInfo, lobbyContext, result) => {
126     Platform.runLater({
127       result match {
128         case GenericResponse(true, _) =>
129 
130           /** If the lobby was successfully created, we link the resulting lobby context and go to the next scene */
131           val multiPlayerLobbyScene = new MultiPlayerLobbyScene(parentStage, listener, () => parentStage.scene = MultiPlayerScene.this, user)
132 
133           /** We link the lobby context */
134           multiPlayerLobbyScene.lobbyContext_=(lobbyContext)
135 
136           /** We link the level info */
137           multiPlayerLobbyScene.levelInfo_=(levelInfo)
138 
139           /** We go to the next scene */
140           parentStage.scene = multiPlayerLobbyScene
141         case GenericResponse(false, _) =>
142           val message = "Can't connect to the server: " +
143             "\n- check that the inserted address is correct" +
144             "\n- check your connection"
145 
146           /** If an error occurred */
147           AlertFactory.createErrorAlert("Error", message).showAndWait()
148       }
149     })
150   }
151

153   root = rootLayout
