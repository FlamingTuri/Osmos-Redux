47     margin = Insets(10.0)
48

51 
52     /** Putting the builder one on top of the other */

62           case EntityType.Sentient => sentientCellEntityCreator.visible = true; sentientCellEntityCreator.entityType_=(EntityType.Sentient)

64           case _ => cellEntityCreator.visible = true; cellEntityCreator.entityType_=(EntityType.Matter)
65         }
66       })
67     }
68

71   /** The placeholder which follows the user mouse and changes appearance on EntityType change */
72   private val _entityPlaceholder: Circle = new Circle() {
73     fill.value = new ImagePattern(ImageLoader.getImage(CellTexture))
74     radius = getVisibleCellBuilder.radius.value

76     /** We set a min and max for the size */

83       case EntityType.Matter => fill.value = new ImagePattern(ImageLoader.getImage(CellTexture))

88       case EntityType.Controlled => fill.value = new ImagePattern(ImageLoader.getImage(ControllerTexture))
89       case _ => fill.value = new ImagePattern(ImageLoader.getImage(CellTexture))
90     })
91   }

109     *
110     * @param event the mouse event

113     entityPlaceholder.centerX.value = event.getX
114     entityPlaceholder.centerY.value = event.getY

117     visibleBuilder.y.value = event.getY - HalfWindowHeight

119 
120   /** This method manages the mouse clicked event, creating a new shape (visible by the user) and a new entity before calling the listener
121     *
122     * @param event the mouse event
123     */
124   def manageMouseClickedEvent(event: MouseEvent): Unit = if (entityPlaceholder.visible.value) {
125     /** Creating the new shape which will be shown to the user */
126     val newShape: Shape = new Circle {
127       fill.value_=(entityPlaceholder.fill.value)
128       centerX = entityPlaceholder.centerX.value
129       centerY = entityPlaceholder.centerY.value
130       radius = entityPlaceholder.radius.value
131       effect.value_=(entityPlaceholder.effect.value)
132     }
133     /** Creating the new editor which will be part of the new level configuration */
134     val newCellEntity = getVisibleCellBuilder create()
135 
136     /** Notify the listener */
137     editorEntityTypeContainerListener.onEntityCreated(newShape, newCellEntity)
138   }
139 
140   /** This method returns the currently visible cell entity creator

144   private def getVisibleCellBuilder: CellEntityCreator = entityBuilders.filter((b) => b.visible.value).head
145 
146   /** Getter which returns the entity placeholder node

148     * @return the entity placeholder node
149     */
150   def entityPlaceholder: Circle = _entityPlaceholder
