21   override def run(): Unit = {
22     status = GameStatus.Running
23 
24     while (!stopFlag) {
25       lock.lock() //blocked if paused
26 
27       val startTick = System.currentTimeMillis()
28       try {
29         //let game progress by updating all systems
30         systems foreach (_.update())
31 
32         //DEBUG ONLY: systems foreach (s => { logRunTime(s.getClass.getSimpleName, () => s.update()) })
33       } finally {
34         tryUnlock()
35       }
36 
37       //game loop iteration must last exactly tickTime, so sleep if the tickTime hasn't been reached yet
38       val execTime = System.currentTimeMillis() - startTick
39 
40       //DEBUG ONLY: Logger.log(s"Execution time: ${execTime}ms")("GameLoop")
41 
42       if (!stopFlag) {
43         val tickTime = getTickTime
44         if (execTime < tickTime) {
45           val sleepTime = tickTime - execTime
46           try {
47             Thread.sleep(sleepTime)
48           } catch {
49             case _: Throwable => //do nothing
50           }
51         } else {
52           Logger.log(s"Exceeded tick time by ${math.abs(tickTime - execTime)}ms (${tickTime}ms - ${engine.getFps}fps)")("GameLoop")
53         }
54       }
55     }
56 
57     status = GameStatus.Stopped
58   }
