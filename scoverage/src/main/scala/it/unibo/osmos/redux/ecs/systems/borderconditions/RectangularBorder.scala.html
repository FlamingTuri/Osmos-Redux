53   override protected def computeNewRadius(): Double = {

57   }
58 
59   private def getPortionOutsideBorder(position: Double, minReachablePosition: Double, maxReachablePosition: Double): Double = {
60     position match {
61       case p if p < minReachablePosition => minReachablePosition - p
62       case p if p > maxReachablePosition => p - maxReachablePosition
63       case _ => 0 // the entity is inside the map edge => radius portion outside the map is 0
64     }
65   }
