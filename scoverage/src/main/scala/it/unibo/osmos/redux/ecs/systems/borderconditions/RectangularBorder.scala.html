25   override def checkAndSolveCollision(entity: CollidableProperty): Unit = {
26     initEntityCollisionData(entity)
27     collisionRule match {
28       case CollisionRules.bouncing =>
29         val newPosition = computeNewPosition()
30         val newSpeed = computeNewSpeed(newPosition)
31         positionComponent.point_(newPosition)
32         speedComponent.vector_(newSpeed)
33       case CollisionRules.instantDeath =>
34         computeNewRadius(position.x, minHorizontalPoint, maxHorizontalPoint)
35         computeNewRadius(position.y, minVerticalPoint, maxVerticalPoint)
36       case _ => throw new IllegalArgumentException
37     }
38   }

40   private def initEntityCollisionData(entity: CollidableProperty): Unit = {

60   private def computeNewPosition(): Point = {

66   private def computeNewSpeed(point: Point): Vector = {
67     val newSpeedX: Double = if (point.x == position.x) speedComponent.vector.x else -speedComponent.vector.x
68     val newSpeedY: Double = if (point.y == position.y) speedComponent.vector.y else -speedComponent.vector.y
69     Vector(newSpeedX, newSpeedY)
70   }

72   private def computeNewRadius(position: Double, minReachablePosition: Double, maxReachablePosition: Double): Unit = {
73     val entityRadius = dimensionComponent.radius
74     position match {
75       case p if p < minReachablePosition => dimensionComponent.radius_(entityRadius - (minReachablePosition - p))
76       case p if p > maxReachablePosition => dimensionComponent.radius_(entityRadius - (p - maxReachablePosition))
77       case _ => // no border collision, do nothing
78     }
79   }
