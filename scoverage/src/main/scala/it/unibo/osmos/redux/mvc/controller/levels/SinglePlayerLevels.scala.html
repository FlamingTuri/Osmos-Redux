8 /** Manages singlePlayer levels. */
9 object SinglePlayerLevels {
10   implicit val who: String = "SinglePlayerLevels"
11   private var levels: List[CampaignLevel] = List()
12 
13   /** Initialize singleton with single player levels.
14     *
15     * @param levelsInfo List of Option[LevelInfo]
16     */
17   def init(levelsInfo: List[Option[LevelInfo]]): Unit = {
18     levels = levelsInfo.flatten.map(lvI => CampaignLevel(lvI.changeAvailability(), CampaignLevelStat()))
19     levels.head.levelInfo.isAvailable = true
20   }
21 
22   /** Return the last unlocked level.
23     *
24     * @return the last unlocked level.
25     */
26   def toDoLevel(campaignLevels: List[CampaignLevel] = levels): String = searchLastAvailableLevel() match {
27     case Some(levelName) => levelName
28     case _ => Logger.log("Error: campaign levels list is empty OR the last level had to be enabled")
29       ""
30   }
31 
32   /** Get the campaign levels info.
33     *
34     * @return List[LevelInfo]
35     */
36   def getLevelsInfo: List[LevelInfo] = levels.map(lv => lv.levelInfo)
37 
38   /** Get the campaign levels.
39     *
40     * @return List[CampaignLevel]
41     */
42   def getCampaignLevels: List[CampaignLevel] = levels
43 
44   /** Should be called when user win or lose a campaign level.
45     *
46     * @param endGame   end game result GameWon or GameLost.
47     * @param levelName name of the played level.
48     */
49   def newEndGameEvent(endGame: GameStateEventWrapper, levelName: String): Unit = endGame match {
50     case GameWon => increaseVictories(levelName)
51       if (toDoLevel().equals(levelName)) unlockNextLevel()
52     case GameLost => increaseDefeats(levelName)
53     case _ => Logger.log("End game state not managed!")
54   }
55 
56   /** Update campaign progress, should be called once when the game starts.
57     *
58     * @param campaignProgress campaign progress List[CampaignLevels]
59     */
60   def updateUserStat(campaignProgress: List[CampaignLevel]): Unit =
61   /* if my values are less updated than the file ones */
62     if (isMostRecent(toDoLevel(campaignProgress))) levels = campaignProgress
63 
64   /** Reset the user progress. */
65   def reset(): Unit = {
66     levels.head.levelStat = CampaignLevelStat()
67     levels.filter(lv => !lv.levelInfo.name.equals(levels.head.levelInfo.name)).foreach(lv => {
68       lv.levelInfo.isAvailable = false
69       lv.levelStat = CampaignLevelStat()
70     })
71     UserProgressFileManager.saveUserProgress(levels)
72     Logger.log("reset done")
73   }
74 
75   /** Unlock the next level. */
76   private def unlockNextLevel(): Unit = levels.map(cLv => cLv.levelInfo).find(lv => !lv.isAvailable) match {
77     case Some(nextLevel) => nextLevel.isAvailable = true
78     case _ => Logger.log("All levels are unlocked")
79   }
80 
81   private def isMostRecent(loadedToDoLevelName: String) =
82     getIndexOfLevelByName(toDoLevel()) <= getIndexOfLevelByName(loadedToDoLevelName)
83 
84   private def getIndexOfLevelByName(levelName: String) = levels.map(cLv => cLv.levelInfo.name).indexOf(levelName)
85 
86   private def searchLastAvailableLevel(cLevels: List[LevelInfo] = getLevelsInfo): Option[String] = cLevels match {
87     case LevelInfo(lv, _, av) :: LevelInfo(_, _, av2) :: _ if av && !av2 => Some(lv)
88     case List(LevelInfo(lv2, _, true)) => Some(lv2)
89     case _ :: t => searchLastAvailableLevel(t)
90     case _ => None
91   }
92 
93   private def findLevelByName(levelName: String): Option[CampaignLevel] = levels.find(cLv => cLv.levelInfo.name.equals(levelName))
94 
95   private def increaseVictories(levelName: String): Unit = findLevelByName(levelName) match {
96     case Some(cLevel: CampaignLevel) => cLevel.levelStat.victories += 1
97     case None => Logger.log("Error: level " + levelName + " doesn't exists [IncreaseVictories]")
98   }
99 
100   private def increaseDefeats(levelName: String): Unit = findLevelByName(levelName) match {
101     case Some(cLevel: CampaignLevel) => cLevel.levelStat.defeats += 1
102     case None => Logger.log("Error: level " + levelName + " doesn't exists [IncreaseDefeats]")
103   }
104 }
