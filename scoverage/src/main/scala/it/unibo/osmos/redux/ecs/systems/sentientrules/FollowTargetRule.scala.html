11 /** Rule to compute the acceleration to follow the target enemy
12   *
13   * @param enemies list of enemies from which choose the target enemy
14   */
15 case class FollowTargetRule(enemies: ListBuffer[SentientEnemyProperty]) extends SentientRule {
16 
17   override def computeRule(sentient: SentientProperty, previousAcceleration: Vector): Vector = {
18     val escapeVelocity = sentient.getSpeedComponent.vector add previousAcceleration
19     findTarget(sentient, enemies, escapeVelocity) match {
20       case Some(target) => followTarget(sentient, target, escapeVelocity)
21       case _ => Vector.zero()
22     }
23   }
24 
25   /** apply a acceleration to the sentient to follow the target
26     *
27     * @param sentient sentient entity
28     * @param target   target entity
29     */
30   private def followTarget(sentient: SentientProperty, target: SentientEnemyProperty, actualVelocity: Vector): Vector = {
31     val nextPositionTarget = target.getPositionComponent.point.add(target.getSpeedComponent.vector)
32     val unitVectorDesiredVelocity = MathUtils.unitVector(nextPositionTarget, sentient.getPositionComponent.point)
33     computeSteer(actualVelocity, unitVectorDesiredVelocity)
34   }
35 
36   /** find the target enemy
37     *
38     * @param sentient sentient entity
39     * @param enemies  list of entity
40     * @return the sentient's enemy with greater target coefficient is present, else None
41     */
42   private def findTarget(sentient: SentientProperty, enemies: ListBuffer[SentientEnemyProperty], escapeVelocity: Vector): Option[SentientEnemyProperty] = {
43     enemies.filter(e => e.getCollidableComponent.isCollidable &&
44       !(e.getTypeComponent.typeEntity == EntityType.AntiMatter) &&
45       sentient.getDimensionComponent.radius > e.getDimensionComponent.radius)
46       .map(e => {
47         if (hasLostRadiusBehaviour(sentient)) {
48           (e, targetCoefficientWithLostRadius(sentient, e, escapeVelocity))
49         } else {
50           (e, targetCoefficientWithoutLostRadius(sentient, e))
51         }
52       })
53       .filter(e => e._2 > 0) match {
54       case list if list.isEmpty => None
55       case list => Some(list.max(Ordering.by((d: (SentientEnemyProperty, Double)) => d._2))._1)
56     }
57   }
58 
59   /** compute the coefficient with lost radius
60     *
61     * @param sentient sentient entity
62     * @param enemy    sentient enemy entity
63     * @return the coefficient representing the radius that can be gained form an enemy
64     */
65   private def targetCoefficientWithLostRadius(sentient: SentientProperty, enemy: SentientEnemyProperty, escapeVelocity: Vector): Double = {
66     val nextPositionTarget = enemy.getPositionComponent.point.add(enemy.getSpeedComponent.vector)
67     val unitVectorDesiredVelocity = MathUtils.unitVector(nextPositionTarget, sentient.getPositionComponent.point)
68     val magnitudeOfRotation = computeUnlimitedSteer(escapeVelocity, unitVectorDesiredVelocity).getMagnitude
69     val lostRadiusPercentage = magnitudeOfRotation * PercentageOfLostRadiusForMagnitudeAcceleration
70     enemy.getDimensionComponent.radius - (sentient.getDimensionComponent.radius * lostRadiusPercentage)
71   }
72 
73   /** compute the coefficient without lost radius
74     *
75     * @param sentient sentient entity
76     * @param enemy    sentient enemy entity
77     * @return a coefficient directly proportional to the enemy's radius and
78     *         inversely proportional to the distance between the entities
79     */
80   private def targetCoefficientWithoutLostRadius(sentient: SentientProperty, enemy: SentientEnemyProperty): Double = {
81     enemy.getDimensionComponent.radius / MathUtils.euclideanDistance(sentient.getPositionComponent, enemy.getPositionComponent)
82   }
83 }
