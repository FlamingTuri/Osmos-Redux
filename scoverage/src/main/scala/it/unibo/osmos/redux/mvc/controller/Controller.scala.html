132   override def initLevel(levelContext: LevelContext, chosenLevel: String, isCustom: Boolean = false): Try[Unit] = {
133     val isSimulation = levelContext.levelContextType == LevelContextType.simulation
134     val loadedLevel: Option[Level] = loadLevel(chosenLevel, isCustom, isSimulation)
135 
136     Try(loadedLevel match {
137       case Some(level) =>
138         //first available player cell is assigned to the user
139         level.entities.find(_.isInstanceOf[PlayerCellEntity]) match {
140           case Some(player) =>
141             //assign current player uuid to the
142             levelContext.setPlayerUUID(player.getUUID)
143             //clear all extra player cells
144             level.entities = level.entities.filterNot(e => e.isInstanceOf[PlayerCellEntity] && e.getUUID != player.getUUID)
145           case None =>
146             if (!isSimulation) throw new IllegalStateException("Cannot start a non-simulation level where no player cell is defined.")
147         }
148 
149         //create the game engine if needed
150         if (engine.isEmpty) engine = Some(GameEngine())
151         //initialize the engine with the chosen level data and the level context
152         engine.get.init(level, levelContext)
153         //signal ui to start the game
154         levelContext.setupLevel(level.levelMap.mapShape)
155       case None =>
156         throw new IllegalArgumentException(s"Unable to load level '$chosenLevel', because its definition was not found.")
157     })
158   }

160   private def loadLevel(chosenLevel: String, isCustom: Boolean, isSimulation: Boolean): Option[Level] =
161     if (isCustom) {
162       /*Because user campaign stats are not influenced by end game results of the custom levels*/
163       lastLoadedLevel = None
164       LevelFileManager.getCustomLevel(chosenLevel)
165     } else {
166       val loadedLevel = LevelFileManager.getLevelFromResource(chosenLevel)
167       if (isSimulation) lastLoadedLevel = None else lastLoadedLevel = Some(chosenLevel)
168       loadedLevel
169     }

171   override def initLobby(user: User, lobbyContext: LobbyContext): Promise[GenericResponse[Boolean]] = {
172     val promise = Promise[GenericResponse[Boolean]]()
173 
174     //subscribe to lobby context to intercept exit from lobby click
175     lobbyContext.subscribe {
176       //if user is defined that the event is from the user and not from the server
177       case LobbyEventWrapper(AbortLobby, Some(_)) =>
178         multiPlayerMode match {
179           case Some(MultiPlayerMode.Server) =>
180             if (server.nonEmpty) {
181               server.get.closeLobby()
182               server.get.kill()
183               multiPlayerMode = None
184               server = None
185             }
186             client = None
187           case Some(MultiPlayerMode.Client) =>
188             if (client.nonEmpty) {
189               client.get.leaveLobby()
190               client.get.kill()
191               multiPlayerMode = None
192               client = None
193             }
194           case _ => //do nothing
195         }
196       case _ => //do nothing
197     }
198 
199     multiPlayerMode = Some(if (user.isServer) MultiPlayerMode.Server else MultiPlayerMode.Client)
200     multiPlayerMode match {
201       case Some(MultiPlayerMode.Server) =>
202         //initialize the server and creates the lobby
203         val server = Server(user.username)
204         server.bind(ActorSystemHolder.createActor(server))
205         //create lobby
206         server.createLobby(lobbyContext)
207         //save server reference
208         this.server = Some(server)
209         promise success GenericResponse(true)
210 
211       case Some(MultiPlayerMode.Client) =>
212         //initialize the client, connects to the server and enters the lobby
213         val client = Client()
214         client.bind(ActorSystemHolder.createActor(client))
215         client.connect(user.ip, user.port).future onComplete {
216           case Success(true) => client.enterLobby(user.username, lobbyContext).future onComplete {
217             case Success(true) =>
218               this.client = Some(client)
219               //creates the level context
220               val levelContext = LevelContext(Constants.MultiPlayer.DefaultClientUUID)
221               //initializes the game
222               client.initGame(levelContext)
223               //fulfill promise
224               promise success GenericResponse(true)
225             case Success(false) => promise success GenericResponse(false, "Unable to enter the lobby, unknown error occurred")
226             case Failure(t) => promise success GenericResponse(false, s"Unable to enter the lobby: ${t.getMessage}")
227           }
228           case Success(false) => promise success GenericResponse(false, "Unable to connect to the server, unknown error")
229           case Failure(t) => promise success GenericResponse(false, s"Unable to connect to the server: ${t.getMessage}")
230         }
231       case _ =>
232         promise success GenericResponse(false, "Cannot initialize the lobby if the multi-player mode is not defined")
233     }
234     promise
235   }

237   override def initMultiPlayerLevel(levelInfo: LevelInfo): Promise[GenericResponse[Boolean]] = {
238     val promise = Promise[GenericResponse[Boolean]]()
239     //End game result of the multiplayer levels doesn't influence campaign statistics
240     lastLoadedLevel = None
241     //load level definition
242     val loadedLevel = LevelFileManager.getLevelFromResource(levelInfo.name, isMultiPlayer = true).get
243 
244     multiPlayerMode.get match {
245       case MultiPlayerMode.Server =>
246         //assign clients to players and wait confirmation
247         server.get.initGame(loadedLevel).future onComplete {
248           case Success(_) =>
249             //create the engine
250             if (engine.isEmpty) engine = Some(GameEngine(Constants.MultiPlayer.DefaultMultiPlayerFps))
251             //initialize the engine and let him create the levelContext
252             val levelContext = engine.get.init(loadedLevel, server.get)
253 
254             //signal server that the game is ready to be started
255             server.get.startGame(levelContext, levelInfo)
256             //tell view to actually start the game
257             levelContext.setupLevel(loadedLevel.levelMap.mapShape)
258 
259             //fulfill the promise
260             promise success GenericResponse(true)
261           case Failure(t) => promise success GenericResponse(false, s"Unable to initialize the game: ${t.getMessage}")
262         }
263       case _ =>
264         promise success GenericResponse(false, "Unable to initialize multi-player level if no lobby have been created.")
265     }
266     promise
267   }
