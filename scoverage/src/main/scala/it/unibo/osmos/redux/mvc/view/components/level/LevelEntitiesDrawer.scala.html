15 /** Class which encapsulates all the required methods and procedures to draw the level entities

33   /** This method let the drawer calculate the new entities colour and then draws them
34     *
35     * @param playerEntity the player entity, which may be empty
36     * @param entities the entities
37     * @param playerCallback the callback called when the player entity gets drawn
38     */
39   def drawEntities(playerEntity: Option[DrawableWrapper], entities: Seq[DrawableWrapper], playerCallback: Point => Unit): Unit = {
40 
41     var entitiesWrappers: Seq[(DrawableWrapper, Color)] = Seq()
42 
43     playerEntity match {
44       /** The player is present */
45       case Some(pe) => entitiesWrappers = calculateColors(entities, pe)
46 
47       /** The player is not present */
48       case _ => entitiesWrappers = calculateColorsWithoutPlayer(entities)
49     }
50 
51     /** We must draw to the screen the entire collection */
52     Platform.runLater({
53       /** Clear the screen */
54       canvas.graphicsContext2D.clearRect(0, 0, canvas.width.value, canvas.height.value)
55       canvas.graphicsContext2D.drawImage(LevelDrawables.backgroundImage, 0, 0, canvas.width.value, canvas.height.value)
56 
57       /** Draw the entities */
58       playerEntity match {
59         case Some(pe) => entitiesWrappers foreach (e => e._1 match {
60           case `pe` =>
61             playerCallback(e._1.center)
62             LevelDrawables.playerCellDrawable.draw(e._1, e._2)
63           case _ => drawEntity(e._1, e._2)
64         })
65         case _ => entitiesWrappers foreach (e => drawEntity(e._1, e._2))
66       }
67     })
68   }

70   /** This method calculates the color of the input entities, interpolating and normalizing it according to the entities size
71     *
72     * @param minColor the base lower Color
73     * @param maxColor the base upper Color
74     * @param entities the input entities
75     * @return the sequence of pair where the first field is the entity and the second is the color
76     */
77   private def calculateColorsWithoutPlayer(entities: Seq[DrawableWrapper], minColor: Color = Color.LightBlue, maxColor: Color = Color.DarkRed): Seq[(DrawableWrapper, Color)] = {
78     entities match {
79       case Nil => Seq()
80       case _ =>
81 
82         /** Calculate the min and max radius among the entities */
83         val endRadius = getEntitiesExtremeRadiusValues(entities)
84 
85         entities map (e => {
86           /** Normalize the entity radius */
87           val normalizedRadius = normalize(e.radius, endRadius._1, endRadius._2)
88 
89           /** Create a pair where the second value is the interpolated color between the two base colors */
90           (e, minColor.interpolate(maxColor, normalizedRadius))
91         }) seq
92     }
93   }
94 
95   /** This method calculates the color of the input entities when the player is present
96     *
97     * @param entities     the input entities
98     * @param playerEntity the player entity
99     * @param minColor     the base lower Color
100     * @param maxColor     the base upper Color
101     * @param playerColor  the player Color
102     * @return the sequence of pair where the first field is the entity and the second is the color
103     */
104   private def calculateColors(entities: Seq[DrawableWrapper], playerEntity: DrawableWrapper,
105                               minColor: Color = ViewConstants.Entities.Colors.DefaultEntityMinColor, maxColor: Color = ViewConstants.Entities.Colors.DefaultEntityMaxColor,
106                               playerColor: Color = Color.Green): Seq[(DrawableWrapper, Color)] = {
107     entities match {
108       case Nil => Seq()
109       case _ =>
110 
111         /** Calculate the min and max radius among the entities, considering the player */
112         entities map {
113           case e if e.radius == playerEntity.radius => (e, playerColor)
114 
115           /** The entity is smaller than the player so it's color hue will approach the min one */
116           case e if e.radius < playerEntity.radius => (e, minColor)
117 
118           /** The entity is larger than the player so it's color hue will approach the max one */
119           case e => (e, maxColor)
120         } seq
121     }
122   }
123 
124   /** Used to draw the correct entity according to its type
125     *
126     * @param drawableWrapper the drawableWrapper
127     * @param color           the border color
128     */
129   private def drawEntity(drawableWrapper: DrawableWrapper, color: Color): Unit = {
130     drawableWrapper.entityType match {
131       case EntityType.Attractive => LevelDrawables.attractiveDrawable.draw(drawableWrapper, color)
132       case EntityType.Repulsive => LevelDrawables.repulsiveDrawable.draw(drawableWrapper, color)
133       case EntityType.AntiMatter => LevelDrawables.antiMatterDrawable.draw(drawableWrapper, color)
134       case EntityType.Sentient => LevelDrawables.sentientDrawable.draw(drawableWrapper, color)
135       case EntityType.Controlled => LevelDrawables.controlledDrawable.draw(drawableWrapper, color)
136       case _ => LevelDrawables.cellDrawable.draw(drawableWrapper, color)
137     }
138   }
139 
140   /** This method returns a pair consisting of the min and the max radius found in the entities sequence
141     *
142     * @param entities a DrawableWrapper sequence
143     * @return a pair consisting of the min and the max radius found; an IllegalArgumentException on empty sequence
144     */
145   private def getEntitiesExtremeRadiusValues(entities: Seq[DrawableWrapper]): (Double, Double) = {
146     /** Sorting the entities */
147     val sorted = entities.sortWith(_.radius < _.radius)
148 
149     /** Retrieving the min and the max radius values */
150     sorted match {
151       case head +: _ :+ tail => (head.radius, tail.radius)
152       case head +: _ => (head.radius, head.radius)
153       case _ => throw new IllegalArgumentException("Could not determine the min and max radius from an empty sequence of entities")
154     }
155   }
