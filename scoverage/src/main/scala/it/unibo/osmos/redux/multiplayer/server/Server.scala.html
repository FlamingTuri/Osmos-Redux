164     override def kill(): Unit = {
165       status match {
166         case ServerState.Game => broadcastMessage(GameEnded(false))
167         case ServerState.Lobby => broadcastMessage(LobbyClosed)
168         case _ => //do nothing is other states
169       }

171       if (lobby.nonEmpty) {
172         lobby = None
173       }
174 
175       if (ref.nonEmpty) {
176         ref.get ! PoisonPill
177         ref = None
178       }

185     override def broadcastMessage(message: Any, clientsToExclude: String*): Unit = {
186       if (ref.isEmpty) throw new IllegalStateException("Unable to broadcast the message, server is not bind to an actor.")
187 
188       val usernameToExclude = clientsToExclude :+ this.username
189       val actors = lobby.get.getPlayers.filterNot(p => (usernameToExclude contains p.getUsername) || !p.isAlive).map(_.getActorRef)
190       actors.foreach(a => a ! message)
191     }

193     override def initGame(level: Level): Promise[Boolean] = {
194       if (status != ServerState.Lobby) throw new UnsupportedOperationException(s"Cannot init the game because the server is in the state: $status")
195 
196       val promise = Promise[Boolean]()
197       //assign player cells to lobby players
198       val futures = setupClients(level)
199       Future.sequence(futures) onComplete {
200         case Success(_) => promise success true
201         case Failure(t) => promise failure t
202       }
203       promise
204     }

208     private def setupClients(level: Level): Seq[Future[Any]] = {
209       val availablePlayerCells = level.entities.filter(_.isInstanceOf[PlayerCellEntity]).map(p => Some(p.getUUID))
210       val otherPlayers = lobby.get.getPlayers.filterNot(_.getUsername == this.username)
211 
212       if (availablePlayerCells.size <= 0) throw new IllegalStateException("Cannot setup clients if the level has no player cells.")
213 
214       //assign first available player cell to the server
215       this.uuid = availablePlayerCells.head.get
216 
217       //update uuid of the server players
218       val serverPlayer = getPlayerFromLobby(this.username)
219       if (serverPlayer.isEmpty) throw new IllegalStateException("Cannot update server player uuid because the player was not found.")
220       serverPlayer.get.setUUID(this.uuid)
221 
222       //get map shape and send to the clients along with the assigned uuid
223       val mapShape = level.levelMap.mapShape
224 
225       //gather extra player cells
226       val remainingPlayerCells = availablePlayerCells.tail
227       val extraCellPlayers = remainingPlayerCells.slice(otherPlayers.size, remainingPlayerCells.size).map(_.get)
228       //update level entities by removing the extra player cells
229       level.entities = for (
230         e <- level.entities
231         if !(extraCellPlayers contains e.getUUID)
232       ) yield e
233 
234       //prepare ask requests for all clients and return futures
235       otherPlayers.map(Some(_)).zipAll(availablePlayerCells.tail, None, None).map {
236         case (Some(p), Some(id)) => p.setUUID(id); Some(p.getActorRef ? GameStarted(id, level.levelInfo, mapShape))
237         case (None, _) => None
238         case _ => throw new IllegalStateException("Not enough player cells for all the clients.")
239       }.filter(_.nonEmpty).map(_.get)
240     }

242     override def startGame(levelContext: MultiPlayerLevelContext, levelInfo: LevelInfo): Unit = {
243       if (status != ServerState.Lobby) throw new UnsupportedOperationException(s"Cannot start game because the server is in the state: $status")
244 
245       lobby match {
246         case Some(gameLobby) =>
247           gameLobby.notifyGameStarted(levelContext, levelInfo)
248           status = ServerState.Game
249         case _ => throw new IllegalArgumentException("Unable to start the game if the lobby is not defined.")
250       }
251     }
252 
253     override def stopGame(winner: String = username): Unit = {
254       if (status != ServerState.Lobby && status != ServerState.Game) throw new UnsupportedOperationException(s"Cannot stop game because the server is in the state: $status")
255 
256       //if the server won, everyone else lost
257       if (winner.equals(username)) {
258         broadcastMessage(ServerActor.GameEnded(false))
259       } else {
260         deliverMessage(winner, GameEnded(true))
261         broadcastMessage(GameEnded(false), winner)
262       }
263       //reset lobby death status
264       resetLobbyPlayersDeathStatus()
265       //change current server status
266       status = ServerState.Lobby
267     }
268 
269     override def deliverMessage(username: String, message: Any): Unit = {
270       if (lobby.isEmpty) throw new UnsupportedOperationException("Cannot deliver message to specific client if the lobby is not defined.")
271 
272       lobby.get.getPlayers.find(_.getUsername == username) match {
273         case Some(player) => player.getActorRef ! message
274         case None => throw new IllegalArgumentException("Cannot deliver message to specific client if the username does not match any player.")
275       }
276     }

278     private def resetLobbyPlayersDeathStatus(): Unit = lobby match {
279       case Some(gameLobby) => gameLobby.getPlayers.foreach(_ setLiveness true)
280       case _ => throw new UnsupportedOperationException("Cannot reset lobby players death status if the lobby is not defined.")
281     }

291     override def removePlayerFromGame(username: String, notify: Boolean = false): Unit = {
292       if (status != ServerState.Game) throw new UnsupportedOperationException(s"Cannot remove player from game because it's in the state: $status")
293 
294       //detect if the dead player is the server itself
295       val isServer = username == this.username
296 
297       //remove entity cell relative to the player that has left
298       val player = getPlayerFromLobby(username)
299       if (player.isEmpty) throw new IllegalArgumentException("Cannot remove player from game because it was not found.")
300 
301       //notify player that he has lost (just clients)
302       if (notify && !isServer) player.get.getActorRef ! GameEnded(false)
303 
304       //get the player entity
305       val playerEntity = EntityManager.filterEntities(classOf[PlayerCellEntity]).find(_.getUUID == player.get.getUUID)
306       if (playerEntity.isEmpty) throw new IllegalArgumentException("Cannot remove player cell from game because it was not found.")
307 
308       EntityManager.delete(playerEntity.get)
309       //set the player as dead only if it's not the server itself
310       setLobbyPlayerAsDead(username)
311     }

313     private def getPlayerFromLobby(username: String): Option[ReferablePlayer] = lobby match {
314       case Some(gameLobby) => gameLobby.getPlayers.find(_.getUsername == username)
315       case _ => throw new UnsupportedOperationException("Cannot get lobby player if the lobby is not defined.")
316     }

318     private def setLobbyPlayerAsDead(username: String): Unit = lobby match {
319       case Some(gameLobby) => gameLobby.getPlayers.filter(_.getUsername == username).foreach(_ setLiveness false)
320       case _ => throw new UnsupportedOperationException("Cannot set lobby player as dead if the lobby is not defined.")
321     }

323     override def createLobby(lobbyContext: LobbyContext): Unit = {
324       if (status != ServerState.Idle) throw new UnsupportedOperationException(s"Server cannot close lobby because it's in the state: $status")
325 
326       //retrieve actor system address
327       val address = ActorSystemHolder.systemAddress
328       //abort if something is wrong with the system
329       if (!NetworkUtils.validatePort(address.port.getOrElse(-1)) ||
330         !NetworkUtils.validateIPV4Address(address.host.getOrElse(""))) {
331         throw new IllegalStateException("Server cannot create the lobby because the actor system gave invalid address or port.")
332       }
333 
334       //Create the lobby
335       lobby = Some(GameLobby(lobbyContext))
336       //add the server itself
337       val serverPlayer = BasePlayer(username, address.host.get, address.port.get)
338       //let interface to show immediately the server player
339       lobbyContext.users = Seq(new User(serverPlayer, true))
340       //add himself to the lobby
341       lobby.get.addPlayer(new ReferablePlayer(serverPlayer, ref.get))
342 
343       status = ServerState.Lobby
344     }

349       if (status != ServerState.Lobby) throw new UnsupportedOperationException(s"Server cannot close lobby because it's in the state: $status")

356       //clear lobby variable
357       lobby = None

359       status = ServerState.Idle
360     }

363       (status, lobby) match {
364         case (ServerState.Lobby | ServerState.Game, Some(gameLobby)) => gameLobby.getPlayers
365         case (_, None) => throw new UnsupportedOperationException(s"Cannot get lobby players because the lobby is not defined.")
366         case _ => throw new UnsupportedOperationException(s"Cannot because it's in the state: $status")
367       }
368     }

370     override def addPlayerToLobby(actorRef: ActorRef, player: BasePlayer): Boolean = {
371       (status, lobby) match {
372         case (ServerState.Lobby, Some(gameLobby)) =>
373           if (!gameLobby.isFull) {
374             val newPlayer = new ReferablePlayer(player.getUsername, player.getAddress, player.getPort, actorRef)
375             //tell everyone a new player entered the lobby
376             broadcastMessage(PlayerEnteredLobby(newPlayer.toBasicPlayer))
377             //add the player to the lobby
378             gameLobby.addPlayer(newPlayer)
379             true
380           } else false
381         case _ => throw new IllegalStateException("Unable to add player to lobby because the lobby is undefined")
382       }
383     }
384 
385     override def removePlayerFromLobby(username: String): Unit = {
386       (status, lobby) match {
387         case (ServerState.Lobby | ServerState.Game, Some(gameLobby)) =>
388           //effectively remove player from lobby
389           gameLobby.removePlayer(username)
390           //tell everyone the player left the game
391           broadcastMessage(PlayerLeftLobby(username))
392         case _ => throw new IllegalStateException("Unable to remove player to lobby because the lobby is undefined")
393       }
394     }
