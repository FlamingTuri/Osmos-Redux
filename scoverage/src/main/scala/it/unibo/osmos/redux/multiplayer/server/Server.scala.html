186     override def broadcastMessage(message: Any, clientsToExclude: String*): Unit = {
187       if (ref.isEmpty) throw new IllegalStateException("Unable to broadcast the message, server is not bind to an actor.")
188 
189       val usernameToExclude = clientsToExclude :+ this.username
190       val actors = lobby.get.getPlayers.filterNot(p => (usernameToExclude contains p.getUsername) || !p.isAlive).map(_.getActorRef)
191       actors.foreach(a => a ! message)

194     override def initGame(level: Level): Promise[Boolean] = {
195       if (status != ServerState.Lobby) throw new UnsupportedOperationException(s"Cannot init the game because the server is in the state: $status")
196 
197       val promise = Promise[Boolean]()
198       //assign player cells to lobby players
199       val futures = setupClients(level)
200       Future.sequence(futures) onComplete {
201         case Success(_) => promise success true
202         case Failure(t) => promise failure t
203       }
204       promise
205     }

209     private def setupClients(level: Level): Seq[Future[Any]] = {
210       val availablePlayerCells = level.entities.filter(_.isInstanceOf[PlayerCellEntity]).map(p => Some(p.getUUID))
211       val otherPlayers = lobby.get.getPlayers.filterNot(_.getUsername == this.username)
212 
213       if (availablePlayerCells.size <= 0) throw new IllegalStateException("Cannot setup clients if the level has no player cells.")
214 
215       //assign first available player cell to the server
216       this.uuid = availablePlayerCells.head.get
217 
218       //update uuid of the server players
219       val serverPlayer = getPlayerFromLobby(this.username)
220       if (serverPlayer.isEmpty) throw new IllegalStateException("Cannot update server player uuid because the player was not found.")
221       serverPlayer.get.setUUID(this.uuid)
222 
223       //get map shape and send to the clients along with the assigned uuid
224       val mapShape = level.levelMap.mapShape
225 
226       //gather extra player cells
227       val remainingPlayerCells = availablePlayerCells.tail
228       val extraCellPlayers = remainingPlayerCells.slice(otherPlayers.size, remainingPlayerCells.size).map(_.get)
229       //update level entities by removing the extra player cells
230       level.entities = for (
231         e <- level.entities
232         if !(extraCellPlayers contains e.getUUID)
233       ) yield e
234 
235       //prepare ask requests for all clients and return futures
236       otherPlayers.map(Some(_)).zipAll(availablePlayerCells.tail, None, None).map {
237         case (Some(p), Some(id)) => p.setUUID(id); Some(p.getActorRef ? GameStarted(id, level.levelInfo, mapShape))
238         case (None, _) => None
239         case _ => throw new IllegalStateException("Not enough player cells for all the clients.")
240       }.filter(_.nonEmpty).map(_.get)
241     }

243     override def startGame(levelContext: MultiPlayerLevelContext, levelInfo: LevelInfo): Unit = {

254     override def stopGame(winner: String = username): Unit = {
255       if (status != ServerState.Lobby && status != ServerState.Game) throw new UnsupportedOperationException(s"Cannot stop game because the server is in the state: $status")
256 
257       //if the server won, everyone else lost
258       if (winner.equals(username)) {
259         broadcastMessage(ServerActor.GameEnded(false))
260       } else {
261         deliverMessage(winner, GameEnded(true))
262         broadcastMessage(GameEnded(false), winner)
263       }
264       //reset lobby death status
265       resetLobbyPlayersDeathStatus()
266       //change current server status
267       status = ServerState.Lobby
268     }

270     override def deliverMessage(username: String, message: Any): Unit = {
271       if (lobby.isEmpty) throw new UnsupportedOperationException("Cannot deliver message to specific client if the lobby is not defined.")
272 
273       lobby.get.getPlayers.find(_.getUsername == username) match {
274         case Some(player) => player.getActorRef ! message
275         case None => throw new IllegalArgumentException("Cannot deliver message to specific client if the username does not match any player.")
276       }

292     override def removePlayerFromGame(username: String, notify: Boolean = false): Unit = {
293       if (status != ServerState.Game) throw new UnsupportedOperationException(s"Cannot remove player from game because it's in the state: $status")
294 
295       //detect if the dead player is the server itself
296       val isServer = username == this.username
297 
298       //remove entity cell relative to the player that has left
299       val player = getPlayerFromLobby(username)
300       if (player.isEmpty) throw new IllegalArgumentException("Cannot remove player from game because it was not found.")
301 
302       //notify player that he has lost (just clients)
303       if (notify && !isServer) player.get.getActorRef ! GameEnded(false)
304 
305       //get the player entity
306       val playerEntity = EntityManager.filterEntities(classOf[PlayerCellEntity]).find(_.getUUID == player.get.getUUID)
307       if (playerEntity.isEmpty) throw new IllegalArgumentException("Cannot remove player cell from game because it was not found.")
308 
309       EntityManager.delete(playerEntity.get)
310       //set the player as dead only if it's not the server itself
311       setLobbyPlayerAsDead(username)
312     }

324     override def createLobby(lobbyContext: LobbyContext): Unit = {
325       if (status != ServerState.Idle) throw new UnsupportedOperationException(s"Server cannot close lobby because it's in the state: $status")
326 
327       //retrieve actor system address
328       val address = ActorSystemHolder.systemAddress
329       //abort if something is wrong with the system
330       if (!NetworkUtils.validatePort(address.port.getOrElse(-1)) ||
331         !NetworkUtils.validateIPV4Address(address.host.getOrElse(""))) {
332         throw new IllegalStateException("Server cannot create the lobby because the actor system gave invalid address or port.")
333       }
334 
335       //Create the lobby
336       lobby = Some(GameLobby(lobbyContext))
337       //add the server itself
338       val serverPlayer = BasePlayer(username, address.host.get, address.port.get)
339       //let interface to show immediately the server player
340       lobbyContext.users = Seq(new User(serverPlayer, true))
341       //add himself to the lobby
342       lobby.get.addPlayer(new ReferablePlayer(serverPlayer, ref.get))
343 
344       status = ServerState.Lobby
345     }

371     override def addPlayerToLobby(actorRef: ActorRef, player: BasePlayer): Boolean = {
372       (status, lobby) match {
373         case (ServerState.Lobby, Some(gameLobby)) =>
374           if (!gameLobby.isFull) {
375             val newPlayer = new ReferablePlayer(player.getUsername, player.getAddress, player.getPort, actorRef)
376             //tell everyone a new player entered the lobby
377             broadcastMessage(PlayerEnteredLobby(newPlayer.toBasicPlayer))
378             //add the player to the lobby
379             gameLobby.addPlayer(newPlayer)
380             true
381           } else false
382         case _ => throw new IllegalStateException("Unable to add player to lobby because the lobby is undefined")
383       }
384     }

386     override def removePlayerFromLobby(username: String): Unit = {
387       (status, lobby) match {

391           //tell everyone the player left the game

393         case _ => throw new IllegalStateException("Unable to remove player to lobby because the lobby is undefined")
394       }
