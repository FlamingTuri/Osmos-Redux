11 /** Rule to compute the acceleration to run away from enemies
12   *
13   * @param enemies list of possible enemies

15 case class EscapeFromEnemiesRule(enemies: ListBuffer[SentientEnemyProperty]) extends SentientRule {

17   /*min value of distance between the entity before divide their unitVector for distance,
18     so the nearest entities have greater weight*/
19   private val minValueOfDistance: Double = 1

21   override def computeRule(sentient: SentientProperty, previousAcceleration: Vector): Vector = {
22     escapeFromEnemies(sentient, findEnemies(sentient, enemies), previousAcceleration)

25   /** search sentient enemies
26     *
27     * @param sentient sentient entity
28     * @param enemies  list of all entities
29     * @return list of sentient's enemies
30     */
31   private def findEnemies(sentient: SentientProperty, enemies: ListBuffer[SentientEnemyProperty]): List[SentientEnemyProperty] =
32     enemies.filter(e => e.getCollidableComponent.isCollidable &&
33       (e.getTypeComponent.typeEntity == EntityType.AntiMatter ||
34         sentient.getDimensionComponent.radius < e.getDimensionComponent.radius)) toList
35 
36   /** compute acceleration to run away from all enemies
37     *
38     * @param sentient sentient entity
39     * @param enemies  list of enemies
40     */
41   private def escapeFromEnemies(sentient: SentientProperty, enemies: List[SentientEnemyProperty], previousAcceleration: Vector): Vector = {
42     val actualSpeed = sentient.getSpeedComponent.vector add previousAcceleration
43     val desiredSeparation = getDesiredSeparation(actualSpeed)
44     enemies.map(e => (e, computeDistance(sentient, e)))
45       .filter(p => p._2 < desiredSeparation)
46       .shiftDistance(minValueOfDistance)
47       .map(m => MathUtils.unitVector(sentient.getPositionComponent.point, m._1.getPositionComponent.point) divide m._2)
48       .foldLeft((Vector.zero(), 1))((acc, i) => (acc._1 add ((i subtract acc._1) divide acc._2), acc._2 + 1))._1 normalized() match {
49       case unitVectorDesiredVelocity if unitVectorDesiredVelocity == Vector(0, 0) => Vector.zero()
50       case unitVectorDesiredVelocity =>
51         computeSteer(actualSpeed, unitVectorDesiredVelocity) multiply WeightOfEscapeAccelerationFromEnemies
52     }
53   }
54 
55   private def computeDistance(sentient: SentientProperty, enemy: SentientEnemyProperty): Double = {
56     val dist = MathUtils.euclideanDistance(sentient.getPositionComponent, enemy.getPositionComponent)
57     dist - sentient.getDimensionComponent.radius - enemy.getDimensionComponent.radius
58   }
