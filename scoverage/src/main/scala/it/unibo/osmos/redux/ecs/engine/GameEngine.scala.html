73   /** The Game engine class implementation.
74     *
75     * @param frameRate The frame rate of the game.
76     */
77   private case class GameEngineImpl(private var frameRate: Int = Constants.Engine.DefaultFps) extends GameEngine {
78 
79     private var gameLoop: Option[GameLoop] = _
80 
81     override def init(level: Level, levelContext: LevelContext): Unit = {
82 
83       //clear all
84       clear()
85 
86       //register InputEventStack to the mouse event listener to collect input events
87       levelContext.subscribe(e => {
88         InputEventQueue.enqueue(e)
89       })
90 
91       //create systems, add to list, the order in this collection is the final system order in the game loop
92       val systems = ListBuffer[System]()
93       if (!(levelContext.levelContextType == LevelContextType.simulation)) systems += InputSystem()
94       systems ++= initMainSystems(level, levelContext)
95       if (!(levelContext.levelContextType == LevelContextType.simulation)) systems += EndGameSystem(levelContext, level.levelInfo.victoryRule)
96 
97 
98       //add all entities in the entity manager (systems are subscribed to EntityManager event when created)
99       level.entities foreach (EntityManager add _)
100 
101       //init the gameloop
102       gameLoop = Some(new GameLoop(this, systems.toList))
103     }
104 
105     override def clear(): Unit = {
106       //clear entities
107       EntityManager.clear()
108       //clear mouse events
109       InputEventQueue.dequeueAll()
110       //reset game speed
111       frameRate = Constants.Engine.DefaultFps
112 
113       gameLoop match {
114         case Some(i) => i.getStatus match {
115           case GameStatus.Running | GameStatus.Paused => throw new IllegalStateException("Unable to clear game engine if the game loop is still running or is paused")
116           case _ => i.kill()
117         }
118         case _ => //do nothing if it's not present
119       }
120       gameLoop = None
121     }
122 
123     /** Initializes the main game systems
124       *
125       * @param level        The level
126       * @param levelContext The level context
127       * @return The list of all main systems
128       */
129     private def initMainSystems(level: Level, levelContext: LevelContext): List[System] = {
130       List(SpawnSystem(), GravitySystem(), MovementSystem(), CollisionSystem(level), CellsEliminationSystem(), SentientSystem(level), DrawSystem(levelContext))
131     }
132 
133     override def init(level: Level, server: Server): MultiPlayerLevelContext = {
134 
135       //clear all
136       clear()
137 
138       //obtain server entity cell uuid
139       val serverPlayer = level.entities.find(_.isInstanceOf[PlayerCellEntity])
140       if (serverPlayer.isEmpty) throw new IllegalArgumentException("Game Engine cannot initialize multi-player game because no player cell entity is present in the level definition.")
141 
142       //create the level context
143       val levelContext = LevelContext(serverPlayer.get.getUUID)
144 
145       //register InputEventQueue to the mouse event listener to collect input events
146       levelContext.subscribe {
147         InputEventQueue enqueue _
148       }
149 
150       //create systems, add to list, the order in this collection is the final system order in the game loop
151       val systems = ListBuffer[System](InputSystem())
152       systems ++= initMainSystems(level, levelContext) :+ MultiPlayerEndGameSystem(server, levelContext, level.levelInfo.victoryRule) :+ MultiPlayerUpdateSystem(server)
153 
154       //add all entities in the entity manager (systems are subscribed to EntityManager event when created)
155       level.entities foreach (EntityManager add _)
156 
157       //init the gameloop
158       gameLoop = Some(new GameLoop(this, systems.toList))
159 
160       levelContext
161     }
162 
163     override def start(): Unit = {
164       gameLoop match {
165         case Some(g) => g.getStatus match {
166           case GameStatus.Idle => g.start()
167           case _ => throw new IllegalStateException("Unable to start game loop because the current game status is not idle.")
168         }
169         case None => throw new IllegalStateException("Unable to start game loop because it hasn't been initialized yet.")
170       }
171     }
172 
173     override def pause(): Unit = {
174       gameLoop match {
175         case Some(g) => g.pause()
176         case None => throw new IllegalStateException("Unable to pause game loop because it hasn't been initialized yet")
177       }
178     }
179 
180     override def resume(): Unit = {
181       gameLoop match {
182         case Some(g) => g.unPause()
183         case None => throw new IllegalStateException("Unable to resume game loop because it hasn't been initialized yet")
184       }
185     }
186 
187     override def stop(): Unit = {
188       gameLoop match {
189         case Some(g) => g.kill()
190         case None => throw new IllegalStateException("Unable to stop game loop because it hasn't been initialized yet")
191       }
192       gameLoop = None
193     }
194 
195     override def getStatus: GameStatus = {
196       gameLoop match {
197         case Some(g) => g.getStatus
198         case None => throw new IllegalStateException("Unable to get game status if the game loop is not present.")
199       }
200     }
201 
202     override def getFps: Int = frameRate
203 
204     override def changeSpeed(increment: Boolean): Unit = {
205       if (increment && frameRate < Constants.Engine.MaximumFps) {
206         frameRate += Constants.Engine.FpsChangeStep
207       } else if (!increment && frameRate > Constants.Engine.MinimumFps) {
208         frameRate -= Constants.Engine.FpsChangeStep
209       }
210       Logger.log(s"changedSpeed: $frameRate")("GameEngine")
211     }
212   }
