26     import it.unibo.osmos.redux.utils.Constants.ResourcesPaths._

31   /** Reads a file from the resources folder.
32     *
33     * @param chosenLevel levels name
34     * @return content of file as Level wrapped into an Option
35     */
36   def getLevelFromResource(chosenLevel: String, isMultiPlayer: Boolean = false): Option[Level] = {
37     import it.unibo.osmos.redux.utils.Constants.ResourcesPaths._
38     val levelsPath = (if (isMultiPlayer) MultiPlayerLevelsPath else SinglePlayerLevelsPath) + chosenLevel + jsonExtension
39     textToLevel(getResourcesFileText(levelsPath)) match {
40       case Success(level) => level.checkCellPosition()
41         Some(level)
42       case Failure(e: Throwable) => Logger.log(e.printStackTrace().toString); None
43     }
44   }

46   private def textToLevel(text: String): Try[Level] = Try(text.parseJson.convertTo[Level])

48   /** Save a custom level into user home.
49     *
50     * @param level Level
51     * @return true if the operation is completed with success
52     */
53   def saveCustomLevel(level: Level): Boolean = {
54     val filePath = processFilePath(level.levelInfo.name)
55     val levelFile = new File(filePath._1.toUri)

68   /** Load level from file saved into user home directory
69     *
70     * @param fileName the name of the file
71     * @return an Option with the required Level if it doesn't fail
72     */
73   def getCustomLevel(implicit fileName: String): Option[Level] =
74     loadFile(LevelsDirectory + fileName + jsonExtension) match {
75       case Some(text) => textToLevel(text) match {
76         case Success(level) => level.checkCellPosition()
77           Some(level)
78         case Failure(_) => Logger.log("Error: convertion of custom level " + fileName + " is failed")(who)
79           None
80       }
81       case _ => None
82     }

84   /** Read from file the custom levels info
85     *
86     * @return if exists a list of LevelInfo
87     */
88   def getCustomLevelsInfo: Try[List[LevelInfo]] = {
89     /*Create directory tree for avoid NullPointer exception later*/
90     createDirectoriesTree(new File(LevelsDirectory + "rnd.txt"))
91     Try(new File(LevelsDirectory).listFiles((_, name) => name.endsWith(jsonExtension))
92       .map(f => loadLevelInfo(f))
93       .filter(optLvl => optLvl.isDefined).map(opt => opt.get).toList)
94   }
95 
96   private def loadLevelInfo(implicit fileName: String): Option[LevelInfo] =

101 
102   private def textToLevelInfo(text: String): Try[LevelInfo] = Try(text.parseJson.convertTo[LevelInfo])
103 
104   /** Read from file into resources the levels info
105     *
106     * @return Option[LevelInfo]
107     */
108   def getResourceLevelInfo(filePath: String): Option[LevelInfo] = {
109     Try(getResourcesFileText(filePath).parseJson.convertTo[LevelInfo]) match {
110       case Success(value) => Some(value)
111       case Failure(_) => Logger.log("File not found into resources: " + filePath)

113     }
114   }
115 
116   private def processFilePath(fileName: String, index: Option[Int] = None): (Path, String) = {
117     val uniqueName = fileName + index.getOrElse("")
118     val path = DefaultFS.getPath(LevelsDirectory + uniqueName + jsonExtension)
119     if (Files.exists(path)) {
120       processFilePath(fileName, if (index.isDefined) Some(index.get + 1) else Some(1))
121     } else {
122       (path, uniqueName)
123     }
124   }
125 
126   private implicit def getFileNameWithoutJsonExtension(file: File): String = file.getName.substring(0, file.getName.length - jsonExtension.length)
