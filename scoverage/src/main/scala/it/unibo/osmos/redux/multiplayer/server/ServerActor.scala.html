12 /** Server actor implementation.
13   *
14   * @param server The server to bind.
15   */
16 class ServerActor(private val server: Server) extends Actor {
17 
18   implicit val who: String = "ServerActor"
19 
20   //manager for clients handshaking
21   private lazy val clientsManager = ClientsManager(System.currentTimeMillis())
22 
23   override def receive: Receive = {
24     case Connect(actorRef: ActorRef) =>
25       //generate a new random id
26       val tempID = clientsManager.nextTempID
27       //add temp actor reference to reply to ask
28       clientsManager.addClient(tempID, sender)
29       //send to the real actor reference request for identification
30       actorRef ! Identify(tempID)
31 
32     case ActorIdentity(id, Some(ref)) =>
33       //get temp actor reference
34       clientsManager.getClient(id.toString) match {
35         case Some(askRef) =>
36           //add the real actor reference to the clients manager
37           clientsManager.addClient(id.toString, ref)
38           //fulfill ask request
39           askRef ! Connected(id.toString)
40         case None =>
41           //reject ask request
42           sender ! Disconnected
43           throw new IllegalStateException("Unable to fulfill ask request because the actor ref was not found.")
44       }
45 
46     case ActorIdentity(id, None) =>
47       clientsManager.getClient(id.toString) match {
48         case Some(askRef) =>
49           //reject ask request
50           askRef ! Disconnected
51         case None =>
52           //reject ask request
53           sender ! Disconnected
54           throw new IllegalStateException("Unable to reject ask request because the actor ref was not found.")
55       }
56 
57     case EnterLobby(tempID, username) =>
58       clientsManager.getClient(tempID) match {
59         case Some(ref) =>
60           //get client info
61           val (address, port) = (ref.path.address.host.get, ref.path.address.port.get)
62           //try to add the player to the lobby
63           if (server.addPlayerToLobby(ref, BasePlayer(username, address, port))) {
64             //send info about the lobby
65             sender ! LobbyInfo(server.getLobbyPlayers.map(_.toBasicPlayer))
66             //add watch to this actor
67             context.watchWith(ref, Disconnect(username))
68           } else sender ! LobbyFull
69         case None =>
70           //handshaking failed
71           sender ! Disconnected
72           throw new IllegalArgumentException(s"Unable to found actor ref for client with tempID: $tempID")
73       }
74 
75     case LeaveLobby(username) => context.unwatch(sender); server.removePlayerFromLobby(username)
76 
77     case PlayerInput(event) => server.notifyClientInputEvent(event)
78 
79     case LeaveGame(username) => context.unwatch(sender); server.removePlayerFromGame(username)
80 
81     case Disconnect(username) =>
82       server.getState match {
83         case ServerState.Lobby => server.removePlayerFromLobby(username)
84         case ServerState.Game => server.removePlayerFromGame(username); server.removePlayerFromLobby(username)
85         case _ => //do nothing
86       }
87 
88     case unknownMessage => Logger.log("Received unknown message: " + unknownMessage)
89   }
90 }
