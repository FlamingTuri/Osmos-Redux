28   def checkCellPosition(): Unit = levelMap.mapShape match {

34   private def rectangularMapCheck(rectangle: Rectangle): Unit = {
35     /*calculate map bound*/
36     var westMiddlePointX = rectangle.center.x - (rectangle.base / 2)
37     var northMiddlePointY = rectangle.center.y - (rectangle.height / 2)
38     //const for translate point if they are negative
39     val kx: Double = if (westMiddlePointX < 0) -westMiddlePointX else 0
40     val ky: Double = if (northMiddlePointY < 0) -northMiddlePointY else 0
41 
42     westMiddlePointX = westMiddlePointX + kx
43     northMiddlePointY = northMiddlePointY + ky
44     val southMiddlePointY = northMiddlePointY + rectangle.height
45     val eastMiddlePointX = westMiddlePointX + rectangle.base
46     entities = entities.filterNot(ent => {
47       //calculate cell point
48       val cellCenter = Point(ent.getPositionComponent.point.x + kx, ent.getPositionComponent.point.y + ky)
49       val topY = cellCenter.y - ent.getDimensionComponent.radius
50       val rightX = cellCenter.x + ent.getDimensionComponent.radius
51       val bottomY = topY + 2 * ent.getDimensionComponent.radius
52       val leftX = rightX - 2 * ent.getDimensionComponent.radius
53       //check if cell is into map
54       !(leftX >= westMiddlePointX && rightX <= eastMiddlePointX) || !(topY >= northMiddlePointY && bottomY <= southMiddlePointY)
55     })
56   }
57 
58   private def circularMapCheck(circle: Circle): Unit =
59     entities = entities.map(ent => (ent, MathUtils.euclideanDistance(ent.getPositionComponent.point,
60       Point(circle.center.x, circle.center.y)) +
61       ent.getDimensionComponent.radius))
62       .filterNot(tup => tup._2 > circle.radius)
63       .map(t => t._1)
