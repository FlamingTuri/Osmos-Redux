10 /** System managing the level's ending rules
11   *
12   * @param levelContext object to notify the view of the end game result
13   * @param victoryRules enumeration representing the level's victory rules
14   */
15 case class EndGameSystem(levelContext: GameStateHolder, victoryRules: VictoryRules.Value) extends AbstractSystem[DeathProperty] {
16 
17   private val victoryCondition = victoryRules match {
18     case VictoryRules.becomeTheBiggest => BecomeTheBiggestVictoryCondition()
19     case VictoryRules.becomeHuge => BecomeHugeVictoryCondition()
20     case VictoryRules.absorbTheAttractors => AbsorbCellsWithTypeVictoryCondition(EntityType.Attractive)
21     case VictoryRules.absorbTheRepulsors => AbsorbCellsWithTypeVictoryCondition(EntityType.Repulsive)
22     case VictoryRules.absorbTheHostileCells => AbsorbCellsWithTypeVictoryCondition(EntityType.Sentient)
23     case _ => throw new NotImplementedError()
24   }
25 
26   override def update(): Unit = {
27     if (levelContext.gameCurrentState == GamePending) {
28       val playerEntity = entities.find(_.getTypeComponent.typeEntity == EntityType.Controlled)
29       if (playerEntity isEmpty) {
30         levelContext.notify(GameLost)
31       } else {
32         if (victoryCondition.check(playerEntity.get, entities)) levelContext.notify(GameWon)
33       }
34     }
35   }
36 }
