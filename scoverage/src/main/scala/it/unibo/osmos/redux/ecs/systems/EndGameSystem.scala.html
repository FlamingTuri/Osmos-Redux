10 /** System managing the level's ending rules
11   *
12   * @param levelContext object to notify the view of the end game result
13   * @param victoryRules enumeration representing the level's victory rules
14   */
15 case class EndGameSystem(levelContext: GameStateHolder, victoryRules: VictoryRules.Value) extends AbstractSystem2[PlayerCellEntity, DeathProperty] {
16 
17   private val victoryCondition = victoryRules match {
18     case VictoryRules.becomeTheBiggest => BecomeTheBiggestVictoryCondition()
19     case VictoryRules.becomeHuge => BecomeHugeVictoryCondition()
20     case VictoryRules.absorbTheRepulsors => AbsorbCellsWithTypeVictoryCondition(EntityType.Repulsive)
21     case VictoryRules.absorbTheHostileCells => AbsorbCellsWithTypeVictoryCondition(EntityType.Sentient)
22     case _ => throw new NotImplementedError()
23   }
24 
25   override def update(): Unit = {
26     if (levelContext.gameCurrentState == GamePending) {
27       if (entities isEmpty) {
28         levelContext.notify(GameLost)
29       } else {
30         entities foreach (playerEntity => {
31           if (victoryCondition.check(playerEntity, entitiesSecondType)) levelContext.notify(GameWon)
32         })
33       }
34     }
35   }
36 }
