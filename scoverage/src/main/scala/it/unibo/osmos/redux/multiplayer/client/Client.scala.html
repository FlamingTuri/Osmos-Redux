127   final case class ClientImpl() extends Client {
128 
129     //temp id useful for handshaking
130     private var tempID: String = _
131     //the username of this specific client
132     private var username: String = _
133     //the uuid of the cell entity that represents this client
134     private var uuid: String = _
135 
136     //the current lobby
137     private var lobby: Option[GameLobby[BasePlayer]] = None
138     //the server actor
139     private var server: Option[ActorRef] = None
140     //the client actor
141     private var ref: Option[ActorRef] = None
142 
143     //the level context of the current level
144     private var levelContext: Option[MultiPlayerLevelContext] = None
145 
146     //MAIN METHODS
147 
148     override def bind(actorRef: ActorRef): Unit = ref = Some(actorRef)
149 
150     override def connect(address: String, port: Int): Promise[Boolean] = {
151       val promise = Promise[Boolean]()
152       //resolve remote server actor path
153       resolveRemotePath(generateRemoteActorPath(address, port)) onComplete {
154         case Success(serverRef) =>
155           server = Some(serverRef)
156           //if the remote actor path is successfully resolved try to connect to the server
157           serverRef ? ClientActor.Connect(ref.get) onComplete {
158             case Success(Connected(id)) => this.tempID = id; promise success true
159             case Success(_) => promise success false
160             case Failure(t) => kill(); promise failure t
161           }
162         case Failure(t) => kill(); promise failure t
163       }
164       promise
165     }
166 
167     override def kill(): Unit = {
168       if (lobby.nonEmpty && server.nonEmpty) {
169         if (uuid.nonEmpty && levelContext.nonEmpty) server.get ! LeaveGame(username)
170         else server.get ! LeaveLobby(username)
171         lobby = None
172       }
173 
174       if (levelContext.nonEmpty) {
175         levelContext = None
176       }
177 
178       if (server.nonEmpty) {
179         server = None
180       }
181 
182       if (ref.nonEmpty) {
183         ref.get ! PoisonPill
184         ref = None
185       }
186     }
187 
188     private def generateRemoteActorPath(address: String, port: Int): String =
189       s"""akka.tcp://${Constants.MultiPlayer.ActorSystemName}@$address:$port/user/${Constants.MultiPlayer.ServerActorName}"""
190 
191     //GAME MANAGEMENT
192 
193     private def resolveRemotePath(remotePath: String): Future[ActorRef] =
194       ActorSystemHolder.getSystem.actorSelection(remotePath) resolveOne()
195 
196     override def initGame(levelContext: MultiPlayerLevelContext): Unit = {
197       if (lobby.isEmpty) throw new IllegalStateException("The player entered no lobby, unable to initialize the game.")
198       if (this.levelContext.nonEmpty) throw new IllegalStateException("The server already hold a level context, unable to initialize the game.")
199 
200       //register client to the mouse event listener to send input events to the server
201       levelContext.subscribe(e => {
202         signalPlayerInput(e)
203       })
204       //save the level context, will be used later to communicate with the view
205       this.levelContext = Some(levelContext)
206     }
207 
208     override def signalPlayerInput(event: MouseEventWrapper): Unit = server match {
209       case Some(serverRef) => serverRef ! PlayerInput(event)
210       case _ => throw new IllegalStateException("Unable to signal player input if the server is not defined")
211     }
212 
213     override def startGame(uuid: String, levelInfo: LevelInfo, mapShape: MapShape): Unit = {
214 
215       if (lobby.isEmpty) throw new IllegalArgumentException("Unable to start the game because the lobby is undefined")
216       if (uuid.equals(Constants.MultiPlayer.DefaultClientUUID)) throw new IllegalArgumentException("Invalid player UUID, the client is not be able to send correct inputs to the server")
217 
218       //save entity uuid
219       this.uuid = uuid
220       //update level context uuid
221       levelContext match {
222         case Some(context) =>
223           context.setPlayerUUID(uuid)
224           //notify lobby that the game is started (prepares the view)
225           lobby.get.notifyGameStarted(levelContext.get, levelInfo)
226           //actually starts the game
227           context.setupLevel(mapShape)
228         case _ => throw new IllegalArgumentException("Unable to start the game because the level context is undefined")
229       }
230     }
231 
232     //INPUT MANAGEMENT
233 
234     override def stopGame(victorious: Boolean): Unit = levelContext match {
235       case Some(context) => context.notify(if (victorious) GameWon else GameLost)
236       case _ => throw new IllegalStateException("Unable to stop the game because the level context is not defined")
237     }
238 
239     //LOBBY
240 
241     override def leaveGame(): Unit = {
242       if (username.isEmpty) throw new IllegalStateException("Unable to leave the game if the player username is undefined.")
243 
244       //send leave game only if it's the user will (a.k.a. the game state is still in pending)
245       levelContext match {
246         case Some(context) => (context.gameCurrentState, server) match {
247           case (GamePending, Some(serverRef)) => serverRef.tell(ClientActor.LeaveGame(username), ref.get)
248           case (_, None) => throw new IllegalStateException("Unable to leave the game if the server is not defined.")
249           case _ => //do nothing
250         }
251         case _ => throw new IllegalStateException("Unable to leave the game if the level context is not defined.")
252       }
253     }
254 
255     override def enterLobby(username: String, lobbyContext: LobbyContext): Promise[Boolean] = {
256 
257       if (lobby.nonEmpty) throw new IllegalStateException("Unable to enter lobby if the client is already entered in another one")
258       if (tempID.isEmpty) throw new IllegalStateException("Unable to enter lobby if the client hasn't yet received the temp id from the server")
259 
260       //Save username locally
261       this.username = username
262 
263       val promise = Promise[Boolean]()
264       server.get ? ClientActor.EnterLobby(tempID, username) onComplete {
265         case Success(LobbyInfo(players)) =>
266           lobby = Some(GameLobby(lobbyContext))
267           //add already present players to the lobby
268           lobby.get.addPlayers(players: _*)
269           //set players list into lobby context to let view draw all already present players
270           lobbyContext.users = players.map(p => new User(p, false))
271           //add watch to server remote actor
272           ref.get ! StartWatching(server.get)
273           //fulfill promise
274           promise success true
275         case Success(Disconnected) | Success(UsernameAlreadyTaken) | Success(LobbyFull) => kill(); promise success false
276         case Success(unknown) => kill(); promise failure new IllegalArgumentException(s"Server unexpected replied to enterLobby with unknown message: $unknown")
277         case Failure(t) => kill(); promise failure t
278       }
279       promise
280     }
281 
282     override def leaveLobby(): Unit = {
283       if (lobby.isEmpty) throw new UnsupportedOperationException("Unable to leave lobby if it's not defined.")
284 
285       server match {
286         case Some(serverRef) => serverRef.tell(LeaveLobby(username), ref.get); closeLobby()
287         case _ => throw new UnsupportedOperationException("Unable to leave lobby if the server is not defined.")
288       }
289     }
290 
291     override def closeLobby(byUser: Boolean = true): Unit = {
292       if (lobby.isEmpty) throw new UnsupportedOperationException("Unable to close lobby if it's not defined.")
293 
294       lobby.get.notifyLobbyClosed(byUser)
295       lobby = None
296     }
297 
298     override def getLobbyPlayers: Seq[BasePlayer] = lobby match {
299       case Some(gameLobby) => gameLobby.getPlayers
300       case _ => throw new UnsupportedOperationException("Unable to retrieve players lobby if no lobby is defined.")
301     }
302 
303     override def addPlayerToLobby(player: BasePlayer): Unit = lobby match {
304       case Some(gameLobby) => gameLobby.addPlayer(player)
305       case _ => throw new UnsupportedOperationException("Unable to add player to lobby if no lobby is defined.")
306     }
307 
308     //OBSERVERS MANAGEMENT
309 
310     override def removePlayerFromLobby(username: String): Unit = lobby match {
311       case Some(gameLobby) => gameLobby.removePlayer(username)
312       case _ => throw new UnsupportedOperationException("Unable to remove player from lobby if no lobby is defined.")
313     }
314 
315     //HELPERS
316 
317     override def notifyRedraw(entities: Seq[DrawableEntity]): Unit = {
318       val player = entities.find(_.getUUID == getUUID)
319       if (player.isEmpty) throw new IllegalArgumentException("Unable to draw entities because the player was not found")
320       levelContext.get.drawEntities(player, entities)
321     }
322 
323     override def getUUID: String = uuid
324   }
