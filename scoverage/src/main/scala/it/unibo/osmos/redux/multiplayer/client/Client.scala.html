123 object Client {
124   def apply(): Client = ClientImpl()
125 
126   final case class ClientImpl() extends Client {
127 
128     //temp id useful for handshaking
129     private var tempID: String = _
130     //the username of this specific client
131     private var username: String = _
132     //the uuid of the cell entity that represents this client
133     private var uuid: String = _
134 
135     //the current lobby
136     private var lobby: Option[GameLobby[BasePlayer]] = None
137     //the server actor
138     private var server: Option[ActorRef] = None
139     //the client actor
140     private var ref: Option[ActorRef] = None
141 
142     //the level context of the current level
143     private var levelContext: Option[MultiPlayerLevelContext] = None
144 
145     //MAIN METHODS
146 
147     override def bind(actorRef: ActorRef): Unit = ref = Some(actorRef)
148 
149     override def connect(address: String, port: Int): Promise[Boolean] = {
150       val promise = Promise[Boolean]()
151       //resolve remote server actor path
152       resolveRemotePath(generateRemoteActorPath(address, port)) onComplete {
153         case Success(serverRef) =>
154           server = Some(serverRef)
155           //if the remote actor path is successfully resolved try to connect to the server
156           serverRef ? ClientActor.Connect(ref.get) onComplete {
157             case Success(Connected(id)) => this.tempID = id; promise success true
158             case Success(_) => promise success false
159             case Failure(t) => kill(); promise failure t
160           }
161         case Failure(t) => kill(); promise failure t
162       }
163       promise
164     }
165 
166     override def kill(): Unit = {
167       if (lobby.nonEmpty && server.nonEmpty) {
168         if (uuid.nonEmpty && levelContext.nonEmpty) server.get ! LeaveGame(username)
169         else server.get ! LeaveLobby(username)
170         lobby = None
171       }
172 
173       if (levelContext.nonEmpty) {
174         levelContext = None
175       }
176 
177       if (server.nonEmpty) {
178         server = None
179       }
180 
181       if (ref.nonEmpty) {
182         ref.get ! PoisonPill
183         ref = None
184       }
185     }
186 
187     private def generateRemoteActorPath(address: String, port: Int): String =
188       s"""akka.tcp://${Constants.MultiPlayer.ActorSystemName}@$address:$port/user/${Constants.MultiPlayer.ServerActorName}"""
189 
190     //GAME MANAGEMENT
191 
192     private def resolveRemotePath(remotePath: String): Future[ActorRef] =
193       ActorSystemHolder.getSystem.actorSelection(remotePath) resolveOne()
194 
195     override def initGame(levelContext: MultiPlayerLevelContext): Unit = {
196       if (lobby.isEmpty) throw new IllegalStateException("The player entered no lobby, unable to initialize the game.")
197       if (this.levelContext.nonEmpty) throw new IllegalStateException("The server already hold a level context, unable to initialize the game.")
198 
199       //register client to the mouse event listener to send input events to the server
200       levelContext.subscribe(e => {
201         signalPlayerInput(e)
202       })
203       //save the level context, will be used later to communicate with the view
204       this.levelContext = Some(levelContext)
205     }
206 
207     override def signalPlayerInput(event: MouseEventWrapper): Unit = server match {
208       case Some(serverRef) => serverRef ! PlayerInput(event)
209       case _ => throw new IllegalStateException("Unable to signal player input if the server is not defined")
210     }
211 
212     override def startGame(uuid: String, levelInfo: LevelInfo, mapShape: MapShape): Unit = {
213 
214       if (lobby.isEmpty) throw new IllegalArgumentException("Unable to start the game because the lobby is undefined")
215       if (uuid.equals(Constants.MultiPlayer.DefaultClientUUID)) throw new IllegalArgumentException("Invalid player UUID, the client is not be able to send correct inputs to the server")
216 
217       //save entity uuid
218       this.uuid = uuid
219       //update level context uuid
220       levelContext match {
221         case Some(context) =>
222           context.setPlayerUUID(uuid)
223           //notify lobby that the game is started (prepares the view)
224           lobby.get.notifyGameStarted(levelContext.get, levelInfo)
225           //actually starts the game
226           context.setupLevel(mapShape)
227         case _ => throw new IllegalArgumentException("Unable to start the game because the level context is undefined")
228       }
229     }
230 
231     //INPUT MANAGEMENT
232 
233     override def stopGame(victorious: Boolean): Unit = levelContext match {
234       case Some(context) => context.notify(if (victorious) GameWon else GameLost)
235       case _ => throw new IllegalStateException("Unable to stop the game because the level context is not defined")
236     }
237 
238     //LOBBY
239 
240     override def leaveGame(): Unit = {
241       if (username.isEmpty) throw new IllegalStateException("Unable to leave the game if the player username is undefined.")
242 
243       //send leave game only if it's the user will (a.k.a. the game state is still in pending)
244       levelContext match {
245         case Some(context) => (context.gameCurrentState, server) match {
246           case (GamePending, Some(serverRef)) => serverRef.tell(ClientActor.LeaveGame(username), ref.get)
247           case (_, None) => throw new IllegalStateException("Unable to leave the game if the server is not defined.")
248           case _ => //do nothing
249         }
250         case _ => throw new IllegalStateException("Unable to leave the game if the level context is not defined.")
251       }
252     }
253 
254     override def enterLobby(username: String, lobbyContext: LobbyContext): Promise[Boolean] = {
255 
256       if (lobby.nonEmpty) throw new IllegalStateException("Unable to enter lobby if the client is already entered in another one")
257       if (tempID.isEmpty) throw new IllegalStateException("Unable to enter lobby if the client hasn't yet received the temp id from the server")
258 
259       //Save username locally
260       this.username = username
261 
262       val promise = Promise[Boolean]()
263       server.get ? ClientActor.EnterLobby(tempID, username) onComplete {
264         case Success(LobbyInfo(players)) =>
265           lobby = Some(GameLobby(lobbyContext))
266           //add already present players to the lobby
267           lobby.get.addPlayers(players: _*)
268           //set players list into lobby context to let view draw all already present players
269           lobbyContext.users = players.map(p => new User(p, false))
270           //add watch to server remote actor
271           ref.get ! StartWatching(server.get)
272           //fulfill promise
273           promise success true
274         case Success(Disconnected) | Success(UsernameAlreadyTaken) | Success(LobbyFull) => kill(); promise success false
275         case Success(unknown) => kill(); promise failure new IllegalArgumentException(s"Server unexpected replied to enterLobby with unknown message: $unknown")
276         case Failure(t) => kill(); promise failure t
277       }
278       promise
279     }
280 
281     override def leaveLobby(): Unit = {
282       if (lobby.isEmpty) throw new UnsupportedOperationException("Unable to leave lobby if it's not defined.")
283 
284       server match {
285         case Some(serverRef) => serverRef.tell(LeaveLobby(username), ref.get); closeLobby()
286         case _ => throw new UnsupportedOperationException("Unable to leave lobby if the server is not defined.")
287       }
288     }
289 
290     override def closeLobby(byUser: Boolean = true): Unit = {
291       if (lobby.isEmpty) throw new UnsupportedOperationException("Unable to close lobby if it's not defined.")
292 
293       lobby.get.notifyLobbyClosed(byUser)
294       lobby = None
295     }
296 
297     override def getLobbyPlayers: Seq[BasePlayer] = lobby match {
298       case Some(gameLobby) => gameLobby.getPlayers
299       case _ => throw new UnsupportedOperationException("Unable to retrieve players lobby if no lobby is defined.")
300     }
301 
302     override def addPlayerToLobby(player: BasePlayer): Unit = lobby match {
303       case Some(gameLobby) => gameLobby.addPlayer(player)
304       case _ => throw new UnsupportedOperationException("Unable to add player to lobby if no lobby is defined.")
305     }
306 
307     //OBSERVERS MANAGEMENT
308 
309     override def removePlayerFromLobby(username: String): Unit = lobby match {
310       case Some(gameLobby) => gameLobby.removePlayer(username)
311       case _ => throw new UnsupportedOperationException("Unable to remove player from lobby if no lobby is defined.")
312     }
313 
314     //HELPERS
315 
316     override def notifyRedraw(entities: Seq[DrawableEntity]): Unit = {
317       val player = entities.find(_.getUUID == getUUID)
318       if (player.isEmpty) throw new IllegalArgumentException("Unable to draw entities because the player was not found")
319       levelContext.get.drawEntities(player, entities)
320     }
321 
322     override def getUUID: String = uuid
323   }
