84 /** Entity manager who manage the system entities and notify events to observers */
85 object EntityManager extends EntityManager with Observable {
86   private var observers: List[ObserverEntry] = List()
87   private var entities: mutable.Set[Property] = mutable.Set()
88 
89   override def add(entity: Property): Unit = {
90     entities += entity
91     notifyEvent(observers, EMEvents.entityCreated(entity))
92   }
93 
94   override def delete(entity: Property): Unit = {
95     entities remove entity
96     notifyEvent(observers, EMEvents.entityDeleted(entity))
97   }
98 
99   override def subscribe(observer: Observer, managedEntities: ManagedEntity): Unit =
100     observers = (observer, managedEntities) :: observers
101 
102   override def filterEntities(entityInterface: EntityManager.ManagedEntity): List[Property] =
103     entities.filter(ent => extendsInterface(ent, entityInterface)).toList
104 
105   def extendsInterface(ent: Property, entityInt: EntityManager.ManagedEntity): Boolean =
106     getInterfaces(ent) contains entityInt
107 
108   def getInterfaces(entity: Property): List[Class[_]] =
109     ClassUtils.getAllInterfaces(entity.getClass).asScala.toList
110 
111   override def clear(): Unit = {
112     observers = List()
113     entities.clear()
114   }
115 
116   @tailrec
117   private def notifyEvent(observers: List[ObserverEntry], emEvent: EntityManagerEvent): Unit =
118     observers match {
119       case Nil =>
120       case (obs, entityInt) :: t if extendsInterface(emEvent.entity, entityInt) =>
121         obs.notify(emEvent); notifyEvent(t, emEvent)
122       case _ :: t => notifyEvent(t, emEvent)
123     }
