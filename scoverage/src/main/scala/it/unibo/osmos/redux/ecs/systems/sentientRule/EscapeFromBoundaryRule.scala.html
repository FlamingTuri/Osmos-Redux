12 /** compute the acceleration to avoid the boundary in according with boundary collision rule
13   *
14   * @param levelInfo level that contain collision rule
15   */
16 case class EscapeFromBoundaryRule(levelInfo: Level) extends SentientRule {
17 
18   private val bounceRule = levelInfo.levelMap.mapShape match {
19     case shape: Rectangle => RectangularBorder(Point(shape.center.x, shape.center.y), CollisionRules.bouncing, shape.base, shape.height)
20     case shape: Circle => CircularBorder(Point(shape.center.x, shape.center.y), CollisionRules.bouncing, shape.radius)
21     case _ => throw new IllegalArgumentException
22   }
23 
24   override def computeRule(sentient: SentientProperty, previousAcceleration: Vector): Vector = {
25     escapeFromBoundary(sentient, previousAcceleration)
26   }
27 
28   //compute the acceleration to avoid the boundary in according with boundary collision rule
29   private def escapeFromBoundary(sentient: SentientProperty, previousAcceleration: Vector): Vector = levelInfo.levelMap.collisionRule match {
30     case CollisionRules.instantDeath =>
31       val actualSpeed = sentient.getSpeedComponent.vector add previousAcceleration
32       val sentientCopy = CellBuilder()
33         .withPosition(sentient.getPositionComponent)
34         .withDimension(sentient.getDimensionComponent.radius + getDesiredSeparation(actualSpeed))
35         .withSpeed(actualSpeed.x, actualSpeed.y).buildSentientEntity()
36       bounceRule.checkAndSolveCollision(sentientCopy)
37       if (sentientCopy.getSpeedComponent.vector == actualSpeed) {
38         Vector.zero()
39       } else {
40         val steer = computeSteer(actualSpeed, sentientCopy.getSpeedComponent.vector normalized())
41         steer multiply WeightOfEscapeAccelerationFromBoundary
42       }
43     case _ => Vector.zero()
44   }
45 }
