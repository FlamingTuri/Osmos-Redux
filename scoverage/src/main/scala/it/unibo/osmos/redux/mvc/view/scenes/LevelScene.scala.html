417   /** This method calculates the color of the input entities, interpolating and normalizing it according to the entities size
418     *
419     * @param minColor the base lower Color
420     * @param maxColor the base upper Color
421     * @param entities the input entities
422     * @return the sequence of pair where the first field is the entity and the second is the color
423     */
424   private def calculateColorsWithoutPlayer(entities: Seq[DrawableWrapper], minColor: Color = Color.LightBlue, maxColor: Color = Color.DarkRed): Seq[(DrawableWrapper, Color)] = {
425     entities match {
426       case Nil => Seq()
427       case _ =>
428 
429         /** Calculate the min and max radius among the entities */
430         val endRadius = getEntitiesExtremeRadiusValues(entities)
431 
432         entities map (e => {
433           /** Normalize the entity radius */
434           val normalizedRadius = normalize(e.radius, endRadius._1, endRadius._2)
435 
436           /** Create a pair where the second value is the interpolated color between the two base colors */
437           (e, minColor.interpolate(maxColor, normalizedRadius))
438         }) seq
439     }
440   }

442   /** This method calculates the color of the input entities when the player is present
443     *
444     * @param entities     the input entities
445     * @param playerEntity the player entity
446     * @param minColor     the base lower Color
447     * @param maxColor     the base upper Color
448     * @param playerColor  the player Color
449     * @return the sequence of pair where the first field is the entity and the second is the color
450     */
451   private def calculateColors(entities: Seq[DrawableWrapper], playerEntity: DrawableWrapper,
452                               minColor: Color = ViewConstants.Entities.Colors.DefaultEntityMinColor, maxColor: Color = ViewConstants.Entities.Colors.DefaultEntityMaxColor,
453                               playerColor: Color = Color.Green): Seq[(DrawableWrapper, Color)] = {
454     entities match {
455       case Nil => Seq()
456       case _ =>
457 
458         /** Calculate the min and max radius among the entities, considering the player */
459         entities map {
460           case e if e.radius == playerEntity.radius => (e, playerColor)
461 
462           /** The entity is smaller than the player so it's color hue will approach the min one */
463           case e if e.radius < playerEntity.radius => (e, minColor)
464 
465           /** The entity is larger than the player so it's color hue will approach the max one */
466           case e => (e, maxColor)
467         } seq
468     }
469   }

471   /** This method returns a pair consisting of the min and the max radius found in the entities sequence
472     *
473     * @param entities a DrawableWrapper sequence
474     * @return a pair consisting of the min and the max radius found; an IllegalArgumentException on empty sequence
475     */
476   private def getEntitiesExtremeRadiusValues(entities: Seq[DrawableWrapper]): (Double, Double) = {
477     /** Sorting the entities */
478     val sorted = entities.sortWith(_.radius < _.radius)
479 
480     /** Retrieving the min and the max radius values */
481     sorted match {
482       case head +: _ :+ tail => (head.radius, tail.radius)
483       case head +: _ => (head.radius, head.radius)
484       case _ => throw new IllegalArgumentException("Could not determine the min and max radius from an empty sequence of entities")
485     }
486   }
