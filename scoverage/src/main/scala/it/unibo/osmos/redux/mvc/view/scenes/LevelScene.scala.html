271   override def onLevelSetup(mapShape: MapShape): Unit = mapShape match {
272     case null => throw new IllegalStateException("Map set was null")
273     case _ =>
274       val center = Point(mapShape.center.x + HalfWindowWidth, mapShape.center.y + HalfWindowHeight)
275       mapShape match {
276         case c: MapShape.Circle => mapBorder = new Circle {
277           centerX = center.x
278           centerY = center.y
279           radius = c.radius
280         }
281         case r: MapShape.Rectangle => mapBorder = new Rectangle {
282           x = center.x - r.base / 2
283           y = center.y - r.height / 2
284           width = r.base
285           height = r.height
286         }
287       }
288 
289       /** Configuring the mapBorder */
290       mapBorder.fill = Color.Transparent
291       mapBorder.stroke = Color.White
292       mapBorder.strokeWidth = 5.0
293       mapBorder.opacity <== canvas.opacity
294       mapBorder.pickOnBounds = false
295       mapBorder.mouseTransparent = true
296       mapBorder.effect = new DropShadow() {
297         color = Color.White
298         radius = 10.0
299       }
300 
301       /** Scale binding (with canvas) */
302       mapBorder.scaleX <== canvas.scaleX
303       mapBorder.scaleY <== canvas.scaleY
304 
305       /** Translate binding (with canvas) */
306       mapBorder.translateX <== (-canvas.scaleX * LevelPlayer.playerPosX)
307       mapBorder.translateY <== (-canvas.scaleY * LevelPlayer.playerPosY)
308 
309       Platform.runLater({
310         /** Starting the level */
311         startLevel()
312       })
313   }

315   override def onDrawEntities(playerEntity: Option[DrawableWrapper], entities: Seq[DrawableWrapper]): Unit = {

320         LevelPlayer.playerPosX.value = 0.0
321         LevelPlayer.playerPosY.value = 0.0

324         LevelPlayer.playerPosX.value = centerPoint.x
325         LevelPlayer.playerPosY.value = centerPoint.y
326         setCameraPivot(-LevelPlayer.playerPosX.value * canvas.getScaleX, -LevelPlayer.playerPosY.value * canvas.getScaleY)
327       }
328     })
329   }
