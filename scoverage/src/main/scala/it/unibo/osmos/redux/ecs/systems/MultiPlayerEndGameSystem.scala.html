11 /** System managing the level's ending rules for multi-player matches
12   *
13   * @param levelContext object to notify the view of the end game result
14   * @param victoryRules enumeration representing the level's victory rules
15   */
16 case class MultiPlayerEndGameSystem(server: Server, levelContext: GameStateHolder, victoryRules: VictoryRules.Value) extends AbstractSystem2[PlayerCellEntity, DeathProperty] {
17 
18   private val victoryCondition = victoryRules match {
19     case VictoryRules.absorbAllOtherPlayers => AbsorbAllOtherPlayersCondition()
20     case _ => throw new NotImplementedError()
21   }
22 
23   override def update(): Unit = {
24     if (isGameRunning) {
25       val (alivePlayers, deadPlayers) = server.getLobbyPlayers.filter(_.isAlive) partition (p => entities.map(_.getUUID) contains p.getUUID)
26       val aliveCells = entitiesSecondType.filterNot(c => deadPlayers.map(_.getUUID) contains c.getUUID)
27 
28       //check
29       for (
30         (player, _, isServer) <- alivePlayers
31           .map(i => (i, entities.find(_.getUUID == i.getUUID), i.getUUID == server.getUUID)) //get player, cell and isServer
32           .filter(i => i._2.nonEmpty && victoryCondition.check(i._2.get, aliveCells)) //filter only players that have won
33         if isGameRunning //when a winner is found do not check other players
34       ) yield {
35         //DEBUG ONLY: Logger.log(s"Victory detected for: ${player.getUsername} - ${player.getUUID} (server: $isServer)")("MultiPlayerEndGameSystem")
36         if (isServer) {
37           server.stopGame()
38           levelContext.notify(GameWon)
39         } else {
40           server.stopGame(player.getUsername)
41           levelContext.notify(GameLost)
42         }
43       }
44 
45       //notify all dead players and remove them
46       if (isGameRunning) {
47         deadPlayers.foreach(p => {
48           //check if the server died
49           val isServer = p.getUsername == server.getUsername
50           //remove player from game and notify it (only if it's not the server itself)
51           server.removePlayerFromGame(p.getUsername, !isServer)
52           //if the server lost show the alternate UI
53           if (isServer) levelContext.notify(GameLostAsServer)
54         })
55       }
56     }
57   }
58 
59   private def isGameRunning: Boolean = levelContext.gameCurrentState == GamePending || levelContext.gameCurrentState == GameLostAsServer
60 }
