11 /** System managing the level's ending rules for multi-player matches
12   *
13   * @param levelContext object to notify the view of the end game result
14   * @param victoryRules enumeration representing the level's victory rules
15   */
16 case class MultiPlayerEndGameSystem(server: Server, levelContext: GameStateHolder, victoryRules: VictoryRules.Value) extends AbstractSystem[DeathProperty] {
17 
18   private val victoryCondition = victoryRules match {
19     case VictoryRules.absorbAllOtherPlayers => AbsorbAllOtherPlayersCondition()
20     case _ => throw new UnsupportedOperationException("Specified victory condition cannot be used in multi-player mode.")
21   }
22 
23   override def update(): Unit = {
24     if (isGameRunning) {
25       val playerEntities = entities.filter(_.getTypeComponent.typeEntity == EntityType.Controlled).map(_.getUUID)
26       val (alivePlayers, deadPlayers) = server.getLobbyPlayers.filter(_.isAlive) partition (p => playerEntities contains p.getUUID)
27       val aliveCells = entities.filterNot(c => deadPlayers.map(_.getUUID) contains c.getUUID)
28 
29       //check
30       for (
31         (player, _, isServer) <- alivePlayers
32           .map(i => (i, entities.find(_.getUUID == i.getUUID), i.getUUID == server.getUUID)) //get player, cell and isServer
33           .filter(i => i._2.nonEmpty && victoryCondition.check(i._2.get, aliveCells)) //filter only players that have won
34         if isGameRunning //when a winner is found do not check other players
35       ) yield {
36         //DEBUG ONLY: Logger.log(s"Victory detected for: ${player.getUsername} - ${player.getUUID} (server: $isServer)")("MultiPlayerEndGameSystem")
37         if (isServer) {
38           server.stopGame()
39           levelContext.notify(GameWon)
40         } else {
41           server.stopGame(player.getUsername)
42           levelContext.notify(GameLost)
43         }
44       }
45 
46       //notify all dead players and remove them
47       if (isGameRunning) {
48         deadPlayers.foreach(p => {
49           //check if the server died
50           val isServer = p.getUsername == server.getUsername
51           //remove player from game and notify it (only if it's not the server itself)
52           server.removePlayerFromGame(p.getUsername, !isServer)
53           //if the server lost show the alternate UI
54           if (isServer) levelContext.notify(GameLostAsServer)
55         })
56       }
57     }
58   }
59 
60   private def isGameRunning: Boolean = levelContext.gameCurrentState == GamePending || levelContext.gameCurrentState == GameLostAsServer
61 }
