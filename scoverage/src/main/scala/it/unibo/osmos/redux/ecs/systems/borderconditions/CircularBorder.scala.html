18   override def checkAndSolveCollision(entity: CollidableProperty): Unit = {
19     if (checkCollision(entity)) {
20       collisionRule match {
21         case CollisionRules.bouncing =>
22           val newPosition = computeNewPosition(entity)

24 
25           val speedComponent = entity.getSpeedComponent

29         case CollisionRules.instantDeath =>

41   private def computeNewPosition(entity: CollidableProperty): Point = {
42     val entityPosition = entity.getPositionComponent.point
43     val entitySpeed = entity.getSpeedComponent.vector
44     val A = levelCenter
45 
46     val B = entityPosition subtract entitySpeed
47     val C = entityPosition
48     val R = levelRadius
49     val r = entity.getDimensionComponent.radius
50 
51     val AB = A subtract B
52     val BC = B subtract C
53     val magnitudeOfAB = AB.getMagnitude
54     val magnitudeOfBC = BC.getMagnitude
55 
56     if (magnitudeOfBC == 0) {
57       C
58     } else {
59       val b = AB.dot(BC) / Math.pow(magnitudeOfBC, 2) * -1
60       val c = (Math.pow(magnitudeOfAB, 2) - Math.pow(R - r, 2)) / Math.pow(magnitudeOfBC, 2)
61       val d = b * b - c
62       var k = b - Math.sqrt(d)
63 
64       if (k < 0) {
65         k = b + Math.sqrt(d)
66       }
67 
68       val BD = C subtract B
69       val newMagnitudeOfBD = BD getNewMagnitude (magnitudeOfBC * k)
70 
71       // D
72       B add newMagnitudeOfBD
73     }
74   }

82   private def computeNewSpeed(currentPosition: Point, speedVector: Vector): Vector = {
83     val v = speedVector
84     val n = currentPosition subtract levelCenter normalized()
85 
86     val u = n multiply (v dot n)
87     val w = v subtract u
88     val vAfter = w subtract u
89     val reflection = vAfter subtract v multiply restitution
90     v add reflection
91   }
