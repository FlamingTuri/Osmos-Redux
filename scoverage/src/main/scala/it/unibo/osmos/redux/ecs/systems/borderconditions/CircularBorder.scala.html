23   override protected def computeNewPosition(): Point = {
24     // For better understanding see: http://gamedev.stackexchange.com/a/29658
25     val A = levelCenter
26     val B = currentPosition subtract entitySpeed
27     val C = currentPosition
28     val R = levelRadius
29     val r = entityRadius
30 
31     val AB = A subtract B
32     val BC = B subtract C
33     val magnitudeOfAB = AB.getMagnitude
34     val magnitudeOfBC = BC.getMagnitude
35 
36     if (magnitudeOfBC == 0) {
37       C
38     } else {
39       val b = AB.dot(BC) / Math.pow(magnitudeOfBC, 2) * -1
40       val c = (Math.pow(magnitudeOfAB, 2) - Math.pow(R - r, 2)) / Math.pow(magnitudeOfBC, 2)
41       val d = b * b - c
42       var k = b - Math.sqrt(d)
43 
44       if (k < 0) {
45         k = b + Math.sqrt(d)
46       }
47 
48       val BD = C subtract B
49       val newMagnitudeOfBD = BD getNewMagnitude (magnitudeOfBC * k)
50 
51       // D
52       B add newMagnitudeOfBD
53     }
54   }

56   override protected def computeNewSpeed(newPosition: Point): Vector = {
57     /* For better understanding see the second answer:
58      * https://stackoverflow.com/questions/573084/how-to-calculate-bounce-angle/573206#573206
59      */
60     val v = entitySpeed
61     val n = newPosition subtract levelCenter normalized()
62     val u = n multiply (v dot n)
63     val w = v subtract u
64     val vAfter = w subtract u
65     val reflection = vAfter subtract v multiply restitution
66     v add reflection
67   }
68 
69   override protected def computeNewRadius(): Double = entityRadius - (currentDistanceFromCenter - maxReachableDistance)

77       case _ =>
78     }
79   }
