20   /** Saves user progress.
21     *
22     * @param userProgress List[CampaignLevel]
23     * @return true if the function is completed with success
24     */
25   def saveUserProgress(userProgress: List[CampaignLevel]): Boolean = {
26     val upFile = new File(DefaultFS.getPath(UserProgressFileName + jsonExtension).toUri)
27     createDirectoriesTree(upFile)
28     saveToFile(upFile, userProgress.toJson.prettyPrint)
29   }
30 
31   /** Loads user progress from file.
32     *
33     * @return List[CampaignLevel]
34     */
35   def loadUserProgress(): List[CampaignLevel] =
36     loadFile(UserProgressFileName + jsonExtension) match {
37       case Some(text) => Try(text.parseJson.convertTo[List[CampaignLevel]]) match {
38         case Success(value) => value
39         case Failure(_) => Logger.log("Error: failed user progress convertion to json")
40           deleteFile(Paths.get(UserProgressFileName + jsonExtension))
41           loadUserProgress()
42       }
43       case _ => saveUserProgress(SinglePlayerLevels.getCampaignLevels)
44         loadUserProgress()
45     }
46 
47   /** Delete user progress file
48     *
49     * @return Try[Unit]
50     */
51   def deleteUserProgress(): Try[Unit] = deleteFile(DefaultFS.getPath(UserProgressFileName + jsonExtension).toAbsolutePath)
