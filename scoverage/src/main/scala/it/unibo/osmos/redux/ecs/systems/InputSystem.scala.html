8 case class InputSystem() extends AbstractSystem[InputProperty] {

10   /** Acceleration coefficient to apply to each input movement */

12   /** The lost mass spawn point offset (starting from the

15   private val lostMassSpawnOffset: Double = 0.1

18   /** The initial velocity of the lost mass */

20 
21   override def update(): Unit = {
22 
23     //retrieve all input events
24     val inputEvents = InputEventQueue.dequeueAll()
25 
26     entities foreach (e => {
27       val accel = e.getAccelerationComponent
28       val pos = e.getPositionComponent
29       val spawner = e.getSpawnerComponent
30       val dim = e.getDimensionComponent
31 
32       inputEvents filter (_.uuid == e.getUUID) foreach (ev => {
33 
34         //compute new acceleration direction
35         val directionAcceleration = MathUtils.unitVector(pos.point, ev.point)
36         //apply acceleration
37         accel.vector_(accel.vector add (directionAcceleration multiply accelerationCoefficient))
38 
39         //create a new spawn action
40         val loseMassAmount = dim.radius * lostMassPercentage
41         val directionVector = MathUtils.unitVector(ev.point, pos.point)
42         val spawnPoint = pos.point add (directionVector multiply (dim.radius + lostMassSpawnOffset + loseMassAmount))
43 
44         //enqueue spawner action
45         spawner.enqueueActions(SpawnAction(
46           PositionComponent(Point(spawnPoint.x, spawnPoint.y)),
47           DimensionComponent(loseMassAmount),
48           SpeedComponent(directionVector multiply lostMassInitialVelocity)))
49 
50         //make entity lose mass
51         dim.radius_(dim.radius - loseMassAmount / 2)
52       })
53     })
54   }
