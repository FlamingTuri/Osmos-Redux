10 /** System that managing the collision with boundary and between cell */
11 case class CollisionSystem(levelInfo: Level) extends AbstractSystem[CollidableProperty] {
12 
13   //the percentage of mass that an entity can acquire from another during a collision in a tick
14   private val massExchangeRate = 0.38
15   //constants that controls how much deceleration is applied to an entity when colliding with another one
16   private val decelerationAmount = 0.01
17   //the initial acceleration vector of a steady entity when a collision occurs
18   private val stillEntityInitialAcceleration = 0.0035
19   //the bouncing rule
20   private val bounceRule = levelInfo.levelMap.mapShape match {
21     case shape: Rectangle =>
22       RectangularBorder(Point(shape.center.x, shape.center.y), levelInfo.levelMap.collisionRule, shape.base, shape.height)
23     case shape: Circle =>
24       CircularBorder(Point(shape.center.x, shape.center.y), levelInfo.levelMap.collisionRule, shape.radius)
25     case _ => throw new IllegalArgumentException("Invalid map shape, unable to initialize bounce rule.")
26   }
27 
28   override def update(): Unit = {
29     //check collision with boundary
30     entities foreach (e => bounceRule.checkAndSolveCollision(e))
31     //check collision other entities
32     for {
33       (e1, xIndex) <- entities.zipWithIndex
34       (e2, yIndex) <- entities.zipWithIndex
35       if xIndex < yIndex //skip useless double checks
36       if e1.getCollidableComponent.isCollidable && e2.getCollidableComponent.isCollidable
37       overlap = computeOverlap(e1, e2)
38       if overlap > 0 //check if they overlap (collide)
39     } yield applyCollisionEffects(e1, e2, overlap)
40   }
41 
42   /** Computes the overlap between two entities.
43     *
44     * @param e1 The first entity
45     * @param e2 The second entity
46     * @return The overlap amount.
47     */
48   private def computeOverlap(e1: CollidableProperty, e2: CollidableProperty): Double = {
49     val tinyRadius = (e1.getDimensionComponent.radius, e2.getDimensionComponent.radius) match {
50       case (r1, r2) if r1 > r2 => r2
51       case (r1, _) => r1
52     }
53     val maxDist = MathUtils.euclideanDistance(e1.getPositionComponent.point, e2.getPositionComponent.point)
54     val currDist = e1.getDimensionComponent.radius + e2.getDimensionComponent.radius
55     currDist - maxDist match {
56       case overlap if overlap <= 0 => 0
57       case overlap if overlap > tinyRadius => tinyRadius
58       case overlap => overlap
59     }
60   }
61 
62   /** Applies collision effects to two entities that collide with each other.
63     *
64     * @param e1      The first entity
65     * @param e2      The second entity
66     * @param overlap The overlap amount
67     */
68   private def applyCollisionEffects(e1: CollidableProperty, e2: CollidableProperty, overlap: Double): Unit = {
69     val (bigEntity, smallEntity) = if (e1.getDimensionComponent.radius > e2.getDimensionComponent.radius) (e1, e2) else (e2, e1)
70 
71     //exchange mass between the two entities
72     exchangeMass(bigEntity, smallEntity, overlap)
73 
74     //compute direction vector
75     val dir = MathUtils.unitVector(bigEntity.getPositionComponent.point, smallEntity.getPositionComponent.point)
76 
77     //apply deceleration to both entities, proportionally to their size
78     accelerateEntity(smallEntity, dir multiply -1)
79     accelerateEntity(bigEntity, dir)
80   }
81 
82   /** Exchanges mass from the small entity to the big one (modify entity radius).
83     *
84     * @param bigEntity   The big entity
85     * @param smallEntity The small entity
86     * @param overlap     The overlap amount
87     */
88   private def exchangeMass(bigEntity: CollidableProperty, smallEntity: CollidableProperty, overlap: Double): Unit = {
89     val bigRadius = bigEntity.getDimensionComponent.radius
90     val tinyRadius = smallEntity.getDimensionComponent.radius
91     //reduce radius of the small entity
92     smallEntity.getDimensionComponent.radius_(tinyRadius - overlap * massExchangeRate)
93 
94     //change radius of the big entity and compute the quantity to move the two entity
95     val quantityToMove = (bigEntity.getTypeComponent.typeEntity, smallEntity.getTypeComponent.typeEntity) match {
96       case (EntityType.AntiMatter, _) | (_, EntityType.AntiMatter) =>
97         bigEntity.getDimensionComponent.radius_(bigRadius - overlap * massExchangeRate)
98         (overlap * (1 - massExchangeRate * 2)) / 2
99       case _ =>
100         val oldSmallArea = MathUtils.circleArea(tinyRadius)
101         val newSmallArea = MathUtils.circleArea(smallEntity.getDimensionComponent.radius)
102         val newBigArea = MathUtils.circleArea(bigEntity.getDimensionComponent.radius) + oldSmallArea - newSmallArea
103         bigEntity.getDimensionComponent.radius_(MathUtils.areaToRadius(newBigArea))
104         limitMaxRadius(bigEntity)
105         (overlap - overlap * massExchangeRate + (bigEntity.getDimensionComponent.radius - bigRadius)) / 2
106     }
107 
108     moveEntitiesAfterCollision(bigEntity, smallEntity, quantityToMove)
109   }
110 
111   /** Limit the radius of the entity to the min dimension of the map
112     *
113     * @param entity the entity
114     */
115   private def limitMaxRadius(entity: CollidableProperty): Unit = {
116     val level = levelInfo.levelMap.mapShape
117     val dimension = entity.getDimensionComponent
118     level match {
119       case map: Rectangle =>
120         if (dimension.radius > map.base / 2) {
121           dimension.radius_(map.base / 2)
122         }
123         if (dimension.radius > map.height / 2) {
124           dimension.radius_(map.height / 2)
125         }
126       case _ =>
127     }
128   }
129 
130   /** move each entity in the opposite direction to the other of quantity to move, and check collision with boundary
131     *
132     * @param entity1        first entity
133     * @param entity2        second entity
134     * @param quantityToMove shift of each entity
135     */
136   private def moveEntitiesAfterCollision(entity1: CollidableProperty, entity2: CollidableProperty, quantityToMove: Double): Unit = {
137     val position1 = entity1.getPositionComponent
138     val position2 = entity2.getPositionComponent
139     val unitVector = MathUtils.unitVector(position1.point, position2.point)
140     position1.point_(position1.point add (unitVector multiply quantityToMove))
141     position2.point_(position2.point add (unitVector multiply (-quantityToMove)))
142     bounceRule.repositionIfOutsideMap(entity1)
143     bounceRule.repositionIfOutsideMap(entity2)
144   }
145 
146   /** Applies acceleration to the input entity.
147     *
148     * @param entity    The entity.
149     * @param direction The vector to apply to the acceleration.
150     */
151   private def accelerateEntity(entity: CollidableProperty, direction: Vector): Unit = {
152     val accel = entity.getAccelerationComponent
153 
154     //gain acceleration even if the entity is still
155     if (accel.vector == Vector(0, 0)) {
156       entity.getAccelerationComponent.vector_(direction multiply stillEntityInitialAcceleration)
157     } else {
158       accel.vector_(accel.vector add (direction multiply decelerationAmount))
159     }
160   }
